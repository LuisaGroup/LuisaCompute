32#define LUISA_MAKE_GLOBAL_DSL_BINARY_OP(op, op_tag_name) \
33 template<typename Lhs, typename Rhs> \
34 requires luisa::compute::any_dsl_v<Lhs, Rhs> && \
35 luisa::is_basic_v<luisa::compute::expr_value_t<Lhs>> && \
36 luisa::is_basic_v<luisa::compute::expr_value_t<Rhs>> \
37 [[nodiscard]] inline auto operator op(Lhs &&lhs, Rhs &&rhs) noexcept { \
38 using namespace std::string_view_literals; \
39 static constexpr auto is_logic_op = #op == "||"sv || #op == "&&"sv; \
40 if constexpr (is_logic_op && luisa::compute::is_dsl_v<Rhs>) { \
41 static_assert( \
42 !luisa::compute::is_scalar_expr_v<Lhs> && \
43 !luisa::compute::is_scalar_expr_v<Rhs>, \
44 "Operator `&&` and `||` are not allowed for DSL scalars, " \
45 "since the short-circuit semantics are not implemented. " \
46 "Please use `&` and `|` to workaround."); \
47 } \
48 static constexpr auto is_bit_op = #op == "&"sv || \
49 #op == "|"sv || \
50 #op == "^"sv; \
51 static constexpr auto is_bool_lhs = luisa::compute::is_boolean_expr_v<Lhs>; \
52 static constexpr auto is_bool_rhs = luisa::compute::is_boolean_expr_v<Rhs>; \
53 if constexpr (is_bit_op) { \
54 static_assert( \
55 (is_bool_lhs && is_bool_rhs) || \
56 (!is_bool_lhs && !is_bool_rhs), \
57 "Mixing boolean and non-boolean operands " \
58 "for `&`, `|`, and `^` is not allowed."); \
59 } \
60 using NormalR = std::remove_cvref_t< \
61 decltype(std::declval<luisa::compute::expr_value_t<Lhs>>() op \
62 std::declval<luisa::compute::expr_value_t<Rhs>>())>; \
63 using R = std::conditional_t<is_bit_op && is_bool_lhs, bool, NormalR>; \
64 return luisa::compute::dsl::def<R>( \
65 luisa::compute::detail::FunctionBuilder::current()->binary( \
66 luisa::compute::Type::of<R>(), \
67 luisa::compute::BinaryOp::op_tag_name, \
68 luisa::compute::detail::extract_expression(std::forward<Lhs>(lhs)), \
69 luisa::compute::detail::extract_expression(std::forward<Rhs>(rhs)))); \
70 }
