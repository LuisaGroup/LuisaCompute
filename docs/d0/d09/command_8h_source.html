71#define LUISA_MAKE_COMMAND_COMMON_CREATE(Cmd) \
72 template<typename... Args> \
73 [[nodiscard]] static auto create(Args &&...args) noexcept { \
74 Clock clock; \
75 auto command = detail::pool_##Cmd().create(std::forward<Args>(args)...); \
76 LUISA_VERBOSE_WITH_LOCATION( \
77 "Created {} in {} ms.", #Cmd, clock.toc()); \
78 return command; \
79 }

340 struct alignas(16) Argument {
341
342 enum struct Tag : uint32_t {

396 struct AccelArgument : Argument {
397 uint64_t handle{};
398 AccelArgument() noexcept : Argument{Tag::ACCEL, 0u} {}
399 AccelArgument(uint32_t vid, uint64_t handle) noexcept
400 : Argument{Tag::ACCEL, vid},
401 handle{handle} {}
402 };
403
404 struct ArgumentBuffer : std::array<std::byte, 2048u> {};
405
406private:
407 uint64_t _handle;
408 Function _kernel;
409 size_t _argument_buffer_size{0u};
410 uint _dispatch_size[3]{};
411 uint32_t _argument_count{0u};
412 ArgumentBuffer _argument_buffer{};
413
414public:
415 explicit ShaderDispatchCommand(uint64_t handle, Function kernel) noexcept;
416 void set_dispatch_size(uint3 launch_size) noexcept;
417 [[nodiscard]] auto handle() const noexcept { return _handle; }
418 [[nodiscard]] auto kernel() const noexcept { return _kernel; }
419 [[nodiscard]] auto argument_count() const noexcept { return static_cast<size_t>(_argument_count); }
420 [[nodiscard]] auto dispatch_size() const noexcept { return uint3(_dispatch_size[0], _dispatch_size[1], _dispatch_size[2]); }
421
422 // Note: encode/decode order:
423 // 1. captured buffers
424 // 2. captured textures
425 // 3. captured texture heaps
426 // 4. captured acceleration structures
427 // 4. arguments
428 void encode_buffer(uint32_t variable_uid, uint64_t handle, size_t offset, Usage usage) noexcept;
429 void encode_texture(uint32_t variable_uid, uint64_t handle, uint32_t level, Usage usage) noexcept;
430 void encode_uniform(uint32_t variable_uid, const void *data, size_t size, size_t alignment) noexcept;
431 void encode_bindless_array(uint32_t variable_uid, uint64_t handle) noexcept;
432 void encode_accel(uint32_t variable_uid, uint64_t handle) noexcept;
433
434 template<typename Visit>
435 void decode(Visit &&visit) const noexcept {
436 auto p = _argument_buffer.data();
437 while (p < _argument_buffer.data() + _argument_buffer_size) {
438 Argument argument{};
439 std::memcpy(&argument, p, sizeof(Argument));
440 switch (argument.tag) {
441 case Argument::Tag::BUFFER: {
442 BufferArgument buffer_argument{};
443 std::memcpy(&buffer_argument, p, sizeof(BufferArgument));
444 visit(argument.variable_uid, buffer_argument);
445 p += sizeof(BufferArgument);
446 break;
447 }
448 case Argument::Tag::TEXTURE: {
449 TextureArgument texture_argument{};
450 std::memcpy(&texture_argument, p, sizeof(TextureArgument));
451 visit(argument.variable_uid, texture_argument);
452 p += sizeof(TextureArgument);
453 break;
454 }
455 case Argument::Tag::UNIFORM: {
456 UniformArgument uniform_argument{};
457 std::memcpy(&uniform_argument, p, sizeof(UniformArgument));
458 p += sizeof(UniformArgument);
459 luisa::span data{p, uniform_argument.size};
460 visit(argument.variable_uid, data);
461 p += uniform_argument.size;
462 break;
463 }
464 case Argument::Tag::BINDLESS_ARRAY: {
465 BindlessArrayArgument bindless_array_argument;
466 std::memcpy(&bindless_array_argument, p, sizeof(BindlessArrayArgument));
467 visit(argument.variable_uid, bindless_array_argument);
468 p += sizeof(BindlessArrayArgument);
469 break;
470 }
471 case Argument::Tag::ACCEL: {
472 AccelArgument accel_argument;
473 std::memcpy(&accel_argument, p, sizeof(AccelArgument));
474 visit(argument.variable_uid, accel_argument);
475 p += sizeof(AccelArgument);
476 break;
477 }
478 default: {
479 LUISA_ERROR_WITH_LOCATION("Invalid argument.");
480 break;
481 }
482 }
483 }
484 }
485 LUISA_MAKE_COMMAND_COMMON(ShaderDispatchCommand)
