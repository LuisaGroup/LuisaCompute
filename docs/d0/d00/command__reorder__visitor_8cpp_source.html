45bool CommandReorderVisitor::Overlap(CommandSource sourceA, CommandSource sourceB) {
46
47 // no Usage::NONE by default
48 if (sourceA.usage == Usage::NONE || sourceB.usage == Usage::NONE) {
49 return false;
50 }
51 if (sourceA.usage == Usage::READ && sourceB.usage == Usage::READ) {
52 return false;
53 }
54
55 if (sourceA.type == sourceB.type) {
56 // the same type
57 return sourceA.handle == sourceB.handle;
58 // if (sourceA.handle != sourceB.handle) {
59 // return false;
60 // }
61 // if (sourceA.offset == size_t(-1) || sourceB.offset == size_t(-1) || sourceA.size == size_t(-1) || sourceB.size == size_t(-1)) {
62 // return true;
63 // }
64 // return (sourceA.offset >= sourceB.offset && sourceA.offset <= sourceB.offset + sourceB.size) ||
65 // (sourceA.offset + sourceA.size >= sourceB.offset && sourceA.offset + sourceA.size <= sourceB.offset + sourceB.size) ||
66 // (sourceB.offset >= sourceA.offset && sourceB.offset <= sourceA.offset + sourceA.size) ||
67 // (sourceB.offset + sourceB.size >= sourceA.offset && sourceB.offset + sourceB.size <= sourceA.offset + sourceA.size);
68 }
69 // sourceA will be set to higher level
70 if (sourceB.type == CommandType::BINDLESS_ARRAY || sourceB.type == CommandType::ACCEL ||
71 (sourceB.type == CommandType::MESH && (sourceA.type == CommandType::BUFFER || sourceA.type == CommandType::TEXTURE))) {
72 std::swap(sourceA, sourceB);
73 }
74 switch (sourceA.type) {
75 case CommandType::ACCEL:
76 // accel - xxx
77 if (sourceB.type == CommandType::MESH) {
78 // accel - mesh
79 return (sourceB.usage == Usage::WRITE || sourceB.usage == Usage::READ_WRITE) &&
80 device->is_mesh_in_accel(sourceA.handle, sourceB.handle);
81 }
82 if (sourceB.type == CommandType::BUFFER) {
83 // accel - buffer
84 return (sourceB.usage == Usage::WRITE || sourceB.usage == Usage::READ_WRITE) &&
85 device->is_buffer_in_accel(sourceA.handle, sourceB.handle);
86 }
87 break;
88 case CommandType::MESH:
89 if (sourceB.type == CommandType::BUFFER) {
90 // mesh - buffer
91 return (sourceB.usage == Usage::WRITE || sourceB.usage == Usage::READ_WRITE) &&
92 (sourceB.handle == device->get_vertex_buffer_from_mesh(sourceA.handle) ||
93 sourceB.handle == device->get_triangle_buffer_from_mesh(sourceA.handle));
94 }
95 break;
96 case CommandType::BINDLESS_ARRAY:
97 if (sourceB.type == CommandType::TEXTURE) {
98 // bindless_array - texture
99 return (sourceB.usage == Usage::WRITE || sourceB.usage == Usage::READ_WRITE) &&
100 device->is_texture_in_bindless_array(sourceA.handle, sourceB.handle);
101 }
102 if (sourceB.type == CommandType::BUFFER) {
103 // bindless_array - buffer
104 return (sourceB.usage == Usage::WRITE || sourceB.usage == Usage::READ_WRITE) &&
105 device->is_buffer_in_bindless_array(sourceA.handle, sourceB.handle);
106 }
107 break;
108 default: break;
109 }
110 return false;
111}

113void CommandReorderVisitor::processNewCommandRelation(CommandReorderVisitor::CommandRelation *commandRelation) noexcept {
114 // check all relations by reversed index if they overlap with the command under processing
115 int insertIndex = 0;
116 for (int i = int(_commandRelationData.size()) - 1; i >= std::max(0, int(_commandRelationData.size()) - windowSize); --i) {
117 for (auto &j : _commandRelationData[i]) {
118 CommandRelation *lastCommandRelation = j;
119 bool overlap = false;
120 // check every condition
121 for (const auto &source : commandRelation->sourceSet) {
122 for (const auto &lastSource : lastCommandRelation->sourceSet)
123 if (Overlap(lastSource, source)) {
124 overlap = true;
125 break;
126 }
127 if (overlap)
128 break;
129 }
130 // if overlapping: add relation, tail command is not tail anymore
131 if (overlap) {
132 insertIndex = i + 1;
133 break;
134 }
135 }
136 if (insertIndex > 0)
137 break;
138 }
139
140 if (insertIndex == _commandRelationData.size()) {
141 auto &&v = _commandRelationData.emplace_back();
142 v.emplace_back(commandRelation);
143 } else
144 _commandRelationData[insertIndex].push_back(commandRelation);
145}

147luisa::vector<CommandList> CommandReorderVisitor::getCommandLists() noexcept {
148 luisa::vector<CommandList> ans;
149 for (auto &i : _commandRelationData) {
150 CommandList commandList;
151 for (auto &j : i) {
152 commandList.append(j->command);
153 }
154 if (!commandList.empty()) {
155 ans.push_back(std::move(commandList));
156 }
157 }
158 LUISA_VERBOSE_WITH_LOCATION("Reordered command list size = {}", ans.size());
159 clear();
160 return ans;
161}

355CommandReorderVisitor::CommandRelation *CommandReorderVisitor::allocate_relation(Command *cmd) {
356 if (pooledRelations.empty()) {
357 return relationPool.create(cmd);
358 }
359 auto v = pooledRelations.back();
360 pooledRelations.pop_back();
361 v->command = cmd;
362 return v;
363}
364void CommandReorderVisitor::deallocate_relation(CommandRelation *v) {
365 v->clear();
366 pooledRelations.emplace_back(v);
367}
368CommandReorderVisitor::~CommandReorderVisitor() {
369 for (auto &&i : pooledRelations) {
370 relationPool.recycle(i);
371 }
372}
