7CommandReorderVisitor::ResourceHandle *CommandReorderVisitor::GetHandle(
8 uint64_t tarGetHandle,
9 ResourceType target_type) {
10 auto func = [&](auto &&map) {
11 auto tryResult = map.TryEmplace(
12 tarGetHandle);
13 auto &&value = tryResult.first.Value();
14 if (tryResult.second) {
15 value = handlePool.New();
16 value->handle = tarGetHandle;
17 value->type = target_type;
18 }
19 return value;
20 };

28size_t CommandReorderVisitor::GetLastLayerWrite(ResourceHandle *handle) {
29 size_t layer = std::max<int64_t>(handle->readLayer + 1, handle->writeLayer + 1);
30 switch (handle->type) {
31 case ResourceType::Buffer:
32 case ResourceType::Texture:
33 if (bindlessMaxLayer >= layer) {
34 for (auto &&i : bindlessMap) {
35 if (device->is_buffer_in_bindless_array(i.first, handle->handle)) {
36 layer = std::max<int64_t>(layer, i.second->readLayer + 1);
37 }
38 }
39 }
40 break;
41 case ResourceType::Mesh:
42 layer = std::max<int64_t>(layer, maxAccelLevel + 1);
43 maxMeshLevel = std::max<int64_t>(layer, maxMeshLevel);
44 break;
45 case ResourceType::Accel:
46 layer = std::max<int64_t>(layer, maxMeshLevel + 1);
47 maxAccelLevel = std::max<int64_t>(layer, maxAccelLevel);
48 break;
49 default: break;
50 }
51 return layer;
52}
