151template<size_t i, size_t n, typename I, typename S, typename B>
152inline void soa_read_impl(I index, S s, B buffers) noexcept {
153 if constexpr (i < n) {
154 s.template get<i>() = std::get<i>(buffers).read(index);
155 soa_read_impl<i + 1u, n>(index, s, buffers);
156 }
157}

159template<size_t i, size_t n, typename I, typename S, typename B>
160inline void soa_write_impl(I index, S s, B buffers) noexcept {
161 if constexpr (i < n) {
162 std::get<i>(buffers).write(index, s.template get<i>());
163 soa_write_impl<i + 1u, n>(index, s, buffers);
164 }

192template<typename T, typename SExpr>
193[[nodiscard]] inline auto eval(SExpr &&s_expr) noexcept {
194 static_assert(is_basic_v<T>, "only basic types are allowed in meta-values");
195 auto type = Type::of<T>();
196 auto meta_value = LiteralExpr::MetaValue{
197 type, luisa::string{std::forward<SExpr>(s_expr)}};
198 auto expr = detail::FunctionBuilder::current()->literal(
199 type, std::move(meta_value));
200 return Var{Expr<T>{expr}};
201}

234template<typename T>
235[[nodiscard]] constexpr auto make_vector_tag() noexcept {
236 if constexpr (std::is_same_v<T, bool2>) {
237 return CallOp::MAKE_BOOL2;
238 } else if constexpr (std::is_same_v<T, bool3>) {
239 return CallOp::MAKE_BOOL3;
240 } else if constexpr (std::is_same_v<T, bool4>) {
241 return CallOp::MAKE_BOOL4;
242 } else if constexpr (std::is_same_v<T, int2>) {
243 return CallOp::MAKE_INT2;
244 } else if constexpr (std::is_same_v<T, int3>) {
245 return CallOp::MAKE_INT3;
246 } else if constexpr (std::is_same_v<T, int4>) {
247 return CallOp::MAKE_INT4;
248 } else if constexpr (std::is_same_v<T, uint2>) {
249 return CallOp::MAKE_UINT2;
250 } else if constexpr (std::is_same_v<T, uint3>) {
251 return CallOp::MAKE_UINT3;
252 } else if constexpr (std::is_same_v<T, uint4>) {
253 return CallOp::MAKE_UINT4;
254 } else if constexpr (std::is_same_v<T, float2>) {
255 return CallOp::MAKE_FLOAT2;
256 } else if constexpr (std::is_same_v<T, float3>) {
257 return CallOp::MAKE_FLOAT3;
258 } else if constexpr (std::is_same_v<T, float4>) {
259 return CallOp::MAKE_FLOAT4;
260 } else {
261 static_assert(always_false_v<T>);
262 }

528template<size_t N, typename T>
529 requires is_scalar_expr_v<T>
530[[nodiscard]] inline auto vectorize(T &&x) noexcept {
531 if constexpr (N == 1u) {
532 return Expr{std::forward<T>(x)};
533 } else if constexpr (N == 2u) {
534 return make_vector2(Expr{std::forward<T>(x)});
535 } else if constexpr (N == 3u) {
536 return make_vector3(Expr{std::forward<T>(x)});
537 } else if constexpr (N == 4u) {
538 return make_vector4(Expr{std::forward<T>(x)});
539 } else {
540 static_assert(always_false_v<T>);
541 }
542}

552template<typename... T>
553struct vectorized_dimension {
554 static constexpr size_t value = 0u;
555};
556
557template<typename T>
558struct vectorized_dimension<T> {
559 static constexpr auto value = vector_expr_dimension_v<T>;
560};
561
562template<typename First, typename... Other>
563struct vectorized_dimension<First, Other...> {
564 static constexpr auto value = std::max(
565 vector_expr_dimension_v<First>,
566 vectorized_dimension<Other...>::value);
567};
568
569template<typename... T>
570constexpr auto vectorized_dimension_v = vectorized_dimension<T...>::value;
571
572template<size_t max, typename T>
573using is_dimension_compatible = std::disjunction<
574 std::bool_constant<vector_expr_dimension_v<T> == max>,
575 std::bool_constant<vector_expr_dimension_v<T> == 1u>>;
576
577template<typename... T>
578concept vector_expr_compatible = std::conjunction<
579 is_dimension_compatible<
580 vectorized_dimension_v<T...>, T>...>::value;
581
583template<typename Scalar, typename... T>
584 requires vector_expr_compatible<T...>
585[[nodiscard]] inline auto make_vector_call(CallOp op, T &&...x) noexcept {
586 constexpr auto N = vectorized_dimension_v<T...>;
587 using V = vectorized_t<Scalar, N>;
588 return def<V>(
589 detail::FunctionBuilder::current()->call(
590 Type::of<V>(), op,
591 {vectorize<N>(std::forward<T>(x)).expression()...}));
592}
