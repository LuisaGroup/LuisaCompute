188template<size_t N, typename... Args>
189class Kernel {
190
191 static_assert(N == 1u || N == 2u || N == 3u);
192 static_assert(std::negation_v<std::disjunction<std::is_pointer<Args>...>>);
193 static_assert(std::negation_v<std::disjunction<std::is_reference<Args>...>>);
194

196 friend struct Kernel1D;
197
198 template<typename...>
199 friend struct Kernel2D;

202 friend struct Kernel3D;

204private:
205 using SharedFunctionBuilder = luisa::shared_ptr<const detail::FunctionBuilder>;
206 SharedFunctionBuilder _builder{nullptr};
207 explicit Kernel(SharedFunctionBuilder builder) noexcept : _builder{std::move(builder)} {}
208
209public:
218 template<typename Def>
219 requires std::negation_v<is_callable<std::remove_cvref_t<Def>>> &&
220 std::negation_v<is_kernel<std::remove_cvref_t<Def>>>
221 Kernel(Def &&def)
222 noexcept {
223 static_assert(std::is_invocable_r_v<void, Def, detail::prototype_to_creation_t<Args>...>);
224 _builder = detail::FunctionBuilder::define_kernel([&def] {
225 detail::FunctionBuilder::current()->set_block_size(detail::kernel_default_block_size<N>());
226 []<size_t... i>(auto &&def, std::index_sequence<i...>) noexcept {
227 using arg_tuple = std::tuple<Args...>;
228 using var_tuple = std::tuple<Var<std::remove_cvref_t<Args>>...>;
229 using tag_tuple = std::tuple<detail::prototype_to_creation_tag_t<Args>...>;
230 auto args = detail::create_argument_definitions<var_tuple, tag_tuple>(std::tuple<>{});
231 static_assert(std::tuple_size_v<decltype(args)> == sizeof...(Args));
232 std::invoke(std::forward<decltype(def)>(def),
233 static_cast<detail::prototype_to_creation_t<
234 std::tuple_element_t<i, arg_tuple>> &&>(std::get<i>(args))...);
235 }
236 (std::forward<Def>(def), std::index_sequence_for<Args...>{});
237 });
238 }
239 [[nodiscard]] const auto &function() const noexcept { return _builder; }
240};
