204template<typename T, bool has_step>
205class ForRange {
206
207 static_assert(is_integral_v<T>);
208
209public:
210 struct ForRangeEnd {};
211
213 class ForRangeIter {
214 private:
215 Expr<T> _begin;
216 Expr<T> _end;
217 Expr<T> _step;
218 const Expression *_var{nullptr};
219 const Expression *_cond{nullptr};
220 ForStmt *_stmt{nullptr};
221 uint _time{0u};
222
223 public:
225 explicit ForRangeIter(Expr<T> begin, Expr<T> end, Expr<T> step) noexcept
226 : _begin{begin}, _end{end}, _step{step} {}
227
229 [[nodiscard]] auto operator*() noexcept {
230 if (_time != 0u) [[unlikely]] {
231 LUISA_ERROR_WITH_LOCATION(
232 "Invalid RangeForIter state (with _time = {}).", _time);
233 }
234 auto f = FunctionBuilder::current();
235 Var var{_begin};
236 _var = var.expression();
237 auto bool_type = Type::of<bool>();
238 _cond = f->binary(bool_type, BinaryOp::LESS, _var, _end.expression());
239 if constexpr (has_step) {
240 // step < 0
241 auto neg_step = f->binary(bool_type, BinaryOp::LESS, _step.expression(), f->literal(Type::of<T>(), T{0}));
242 // ((step < 0) && !(var < end)) || (!(step < 0) && (var < end))
243 _cond = f->binary(bool_type, BinaryOp::BIT_XOR, _cond, neg_step);
244 }
245 _stmt = f->for_(_var, _cond, _step.expression());
246 f->push_scope(_stmt->body());
247 return Var{std::move(var)};// to guarantee rvo
248 }
249
250 auto &operator++() noexcept {
251 _time++;
252 FunctionBuilder::current()->pop_scope(_stmt->body());
253 return *this;
254 }
255
256 [[nodiscard]] auto operator!=(ForRangeEnd) const noexcept {
257 return _time == 0u;
258 }
259 };
260
261private:
262 Expr<T> _begin;
263 Expr<T> _end;
264 Expr<T> _step;
265
266public:
267 explicit ForRange(Expr<T> begin, Expr<T> end, Expr<T> step) noexcept
268 : _begin{begin}, _end{end}, _step{step} {}
269 [[nodiscard]] auto begin() const noexcept { return ForRangeIter{_begin, _end, _step}; }
270 [[nodiscard]] auto end() const noexcept { return ForRangeEnd{}; }
271};

289template<typename Lhs, typename Rhs>
290inline void assign(Lhs &&lhs, Rhs &&rhs) noexcept {
291 static_assert(
292 std::tuple_size_v<linear_layout_t<expr_value_t<Lhs>>> ==
293 std::tuple_size_v<linear_layout_t<expr_value_t<Rhs>>>);
294 if constexpr (concepts::assignable<expr_value_t<Lhs>, expr_value_t<Rhs>>) {
295 detail::FunctionBuilder::current()->assign(
296 detail::extract_expression(std::forward<Lhs>(lhs)),
297 detail::extract_expression(std::forward<Rhs>(rhs)));
298 } else if (is_tuple_v<std::remove_cvref_t<Rhs>>) {
299 assign(
300 detail::Ref{std::forward<Lhs>(lhs)},
301 compose(std::forward<Rhs>(rhs)));
302 } else {
303 detail::assign_impl(
304 detail::Ref{std::forward<Lhs>(lhs)},
305 Expr{std::forward<Rhs>(rhs)});
306 }
307}

386template<concepts::iterable AllTags, typename Tag, typename IndexedCase, typename Otherwise>
387 requires concepts::invocable<IndexedCase, int> && concepts::invocable<Otherwise>
388inline void match(AllTags &&tags, Tag &&tag, IndexedCase &&indexed_case, Otherwise &&otherwise) noexcept {
389 auto s = switch_(std::forward<Tag>(tag));
390 auto index = 0;
391 for (auto &&t : std::forward<AllTags>(tags)) {
392 s = std::move(s).case_(t, [&c = indexed_case, i = index] { c(i); });
393 index++;
394 }
395 std::move(s).default_(std::forward<Otherwise>(otherwise));
396}
