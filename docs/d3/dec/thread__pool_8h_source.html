61 template<typename F>
62 requires std::is_invocable_v<F>
63 auto async(F f) noexcept {
64 using R = std::invoke_result_t<F>;
65 auto promise = luisa::make_shared<std::promise<R>>(
66 std::allocator_arg, luisa::allocator{});
67 auto future = promise->get_future().share();
68 _task_count.fetch_add(1u);
69 _dispatch([func = std::move(f), promise = std::move(promise), future, this]() mutable noexcept {
70 if constexpr (std::same_as<R, void>) {
71 func();
72 promise->set_value();
73 } else {
74 promise->set_value(func());
75 }
76 _task_count.fetch_sub(1u);
77 });
78 return future;
79 }

82 template<typename F>
83 requires std::is_invocable_v<F, uint>
84 void parallel(uint n, F f) noexcept {
85 if (n > 0u) {
86 _task_count.fetch_add(1u);
87 auto counter = luisa::make_shared<std::atomic_uint>(0u);
88 _dispatch_all(
89 [=, this]() mutable noexcept {
90 auto i = 0u;
91 while ((i = counter->fetch_add(1u)) < n) { f(i); }
92 if (i == n) { _task_count.fetch_sub(1u); }
93 },
94 n);
95 }
96 }
