208class MemberExpr final : public Expression {
209 friend class AstSerializer;
210
211public:
212 static constexpr auto swizzle_mask = 0xff00000000ull;
213 static constexpr auto swizzle_shift = 32u;

215private:

218
219protected:
220 void _mark(Usage usage) const noexcept override { _self->mark(usage); }

225public:
233 MemberExpr(const Type *type, const Expression *self, size_t member_index) noexcept
234 : Expression{Tag::MEMBER, type}, _self{self}, _member{member_index} {}
235
252 MemberExpr(const Type *type, const Expression *self, size_t swizzle_size, uint64_t swizzle_code) noexcept
253 : Expression{Tag::MEMBER, type}, _self{self}, _member{(static_cast<uint64_t>(swizzle_size) << swizzle_shift) | swizzle_code} {}
254
255 [[nodiscard]] auto is_swizzle() const noexcept { return (_member & swizzle_mask) != 0u; }
256 [[nodiscard]] auto self() const noexcept { return _self; }
257
258 [[nodiscard]] auto member_index() const noexcept {
259 if (is_swizzle()) {
260 LUISA_ERROR_WITH_LOCATION(
261 "Invalid member index in swizzled MemberExpr.");
262 }
263 return static_cast<size_t>(_member);
264 }
265
266 [[nodiscard]] auto swizzle_size() const noexcept {
267 auto s = (_member & swizzle_mask) >> swizzle_shift;
268 if (s == 0u || s > 4u) { LUISA_ERROR_WITH_LOCATION("Invalid swizzle size {}.", s); }
269 return static_cast<size_t>(s);
270 }
271
272 [[nodiscard]] auto swizzle_index(size_t index) const noexcept {
273 if (auto s = swizzle_size(); index >= s) {
274 LUISA_ERROR_WITH_LOCATION(
275 "Invalid swizzle index {} (count = {}).",
276 index, s);
277 }
278 return static_cast<size_t>((_member >> (index * 4u)) & 0x0fu);
279 }
