26template<typename T>
27class Image final : public Resource {
28
29 static_assert(std::disjunction_v<
30 std::is_same<T, int>,
31 std::is_same<T, uint>,
32 std::is_same<T, float>>);

34private:

36 uint32_t _mip_levels{};
37 PixelStorage _storage{};

39private:
40 friend class Device;
41 Image(Device::Interface *device, PixelStorage storage, uint2 size, uint mip_levels = 1u) noexcept
42 : Resource{
43 device,
44 Tag::TEXTURE,
45 device->create_texture(
46 pixel_storage_to_format<T>(storage), 2u,
47 size.x, size.y, 1u,
48 detail::max_mip_levels(make_uint3(size, 1u), mip_levels))},
49 _size{size}, _mip_levels{detail::max_mip_levels(make_uint3(size, 1u), mip_levels)}, _storage{storage} {}

51public:

55 [[nodiscard]] auto native_handle() const noexcept { return device()->texture_native_handle(handle()); }

57 [[nodiscard]] auto size() const noexcept { return _size; }
58 [[nodiscard]] auto mip_levels() const noexcept { return _mip_levels; }
59 [[nodiscard]] auto storage() const noexcept { return _storage; }
60 [[nodiscard]] auto format() const noexcept { return pixel_storage_to_format<T>(_storage); }
61
62 [[nodiscard]] auto view(uint32_t level = 0u) const noexcept {
63 if (level >= _mip_levels) [[unlikely]] {

66 level, _mip_levels);

68 auto mip_size = luisa::max(_size >> level, 1u);

95class ImageView {

97private:

104private:

115 _size{size},

120 [[nodiscard]] auto _as_mipmap() const noexcept {
121 return detail::MipmapView{

123 make_uint3(_size, 1u),
