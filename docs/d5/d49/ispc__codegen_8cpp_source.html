15void ISPCCodegen::visit(const UnaryExpr *expr) {
16 switch (expr->op()) {
17 case UnaryOp::PLUS: _scratch << "unary_plus("; break;
18 case UnaryOp::MINUS: _scratch << "unary_minus("; break;
19 case UnaryOp::NOT: _scratch << "unary_not("; break;
20 case UnaryOp::BIT_NOT: _scratch << "unary_bit_not("; break;
21 default: break;
22 }
23 expr->operand()->accept(*this);
24 _scratch << ")";

27void ISPCCodegen::visit(const BinaryExpr *expr) {
28 if (expr->lhs()->type()->is_scalar() &&
29 expr->rhs()->type()->is_scalar()) {
30 _scratch << "(";
31 expr->lhs()->accept(*this);
32 switch (expr->op()) {
33 case BinaryOp::ADD: _scratch << " + "; break;
34 case BinaryOp::SUB: _scratch << " - "; break;
35 case BinaryOp::MUL: _scratch << " * "; break;
36 case BinaryOp::DIV: _scratch << " / "; break;
37 case BinaryOp::MOD: _scratch << " % "; break;
38 case BinaryOp::BIT_AND: _scratch << " & "; break;
39 case BinaryOp::BIT_OR: _scratch << " | "; break;
40 case BinaryOp::BIT_XOR: _scratch << " ^ "; break;
41 case BinaryOp::SHL: _scratch << " << "; break;
42 case BinaryOp::SHR: _scratch << " >> "; break;
43 case BinaryOp::AND: _scratch << " && "; break;
44 case BinaryOp::OR: _scratch << " || "; break;
45 case BinaryOp::LESS: _scratch << " < "; break;
46 case BinaryOp::GREATER: _scratch << " > "; break;
47 case BinaryOp::LESS_EQUAL: _scratch << " <= "; break;
48 case BinaryOp::GREATER_EQUAL: _scratch << " >= "; break;

83void ISPCCodegen::visit(const MemberExpr *expr) {
84 if (expr->is_swizzle()) {
85 static constexpr std::string_view xyzw[]{"x", "y", "z", "w"};
86 if (auto ss = expr->swizzle_size(); ss == 1u) {
87 expr->self()->accept(*this);
88 _scratch << "._";
89 _scratch << xyzw[expr->swizzle_index(0)];
90 } else {
91 _scratch << "make_";
92 auto elem = expr->type()->element();
93 switch (elem->tag()) {
94 case Type::Tag::BOOL: _scratch << "char"; break;
95 case Type::Tag::INT: _scratch << "int"; break;
96 case Type::Tag::UINT: _scratch << "uint"; break;
97 case Type::Tag::FLOAT: _scratch << "float"; break;
98 default: LUISA_ERROR_WITH_LOCATION(
99 "Invalid vector element type: {}.",
100 elem->description());
101 }
102 _scratch << ss << "(";
103 for (auto i = 0u; i < ss; i++) {
104 expr->self()->accept(*this);
105 _scratch << "._" << xyzw[expr->swizzle_index(i)] << ", ";
106 }
107 _scratch.pop_back();
108 _scratch.pop_back();
109 _scratch << ")";
110 }
111 } else {
112 expr->self()->accept(*this);
113 _scratch << ".m" << expr->member_index();
114 }
115}

117void ISPCCodegen::visit(const AccessExpr *expr) {
118 if (auto t = expr->range()->type(); t->is_array()) {
119 _scratch << "array_access(";
120 } else if (t->is_vector()) {
121 _scratch << "vector_access(";
122 } else if (t->is_matrix()) {
123 _scratch << "matrix_access(";
124 } else if (t->is_buffer()) {
125 _scratch << "buffer_access(";
126 } else {
127 LUISA_ERROR_WITH_LOCATION(
128 "Invalid range type for access expression: {}.",
129 t->description());
130 }
131 expr->range()->accept(*this);
132 _scratch << ", ";
133 expr->index()->accept(*this);
134 _scratch << ")";
135}

139class LiteralPrinter {
140
141private:
142 Codegen::Scratch &_s;
143
144public:
145 explicit LiteralPrinter(Codegen::Scratch &s) noexcept : _s{s} {}
146 void operator()(bool v) const noexcept { _s << v; }
147 void operator()(float v) const noexcept {
148 if (std::isnan(v)) [[unlikely]] { LUISA_ERROR_WITH_LOCATION("Encountered with NaN."); }
149 if (std::isinf(v)) {
150 _s << (v < 0.0f ? "(1.0f/-0.0f)" : "(1.0f/+0.0f)");
151 } else {
152 _s << v << "f";
153 }
154 }
155 void operator()(int v) const noexcept { _s << v; }
156 void operator()(uint v) const noexcept { _s << v << "u"; }
157
158 template<typename T, size_t N>
159 void operator()(Vector<T, N> v) const noexcept {
160 auto t = Type::of<T>();
161 _s << "make_" << t->description() << N << "(";
162 for (auto i = 0u; i < N; i++) {
163 (*this)(v[i]);
164 _s << ", ";
165 }
166 _s.pop_back();
167 _s.pop_back();
168 _s << ")";
169 }
170
171 void operator()(float2x2 m) const noexcept {
172 _s << "make_float2x2(";
173 for (auto col = 0u; col < 2u; col++) {
174 for (auto row = 0u; row < 2u; row++) {
175 (*this)(m[col][row]);
176 _s << ", ";
177 }
178 }
179 _s.pop_back();
180 _s.pop_back();
181 _s << ")";
182 }

450 _scratch << "if (";
451 stmt->condition()->accept(*this);

453 stmt->true_branch()->accept(*this);

460 fb->accept(*this);
461 }

507void ISPCCodegen::_emit_function(Function f) noexcept {
508
509 if (auto iter = std::find(_generated_functions.cbegin(), _generated_functions.cend(), f);
510 iter != _generated_functions.cend()) { return; }
511 _generated_functions.emplace_back(f);
512
513 for (auto &&callable : f.custom_callables()) {
514 _emit_function(callable->function());
515 }
516
517 _continue_analysis.analyze(f);
518 _definition_analysis.analyze(f);
519
520 _function = f;
521 _indent = 0u;
522
523 // constants
524 if (!f.constants().empty()) {
525 for (auto c : f.constants()) { _emit_constant(c); }
526 _scratch << "\n";
527 }
528
529 // signature
530 if (f.tag() == Function::Tag::KERNEL) {
531 _scratch << "static void "
532 << "kernel_"
533 << hash_to_string(f.hash());
534 } else if (f.tag() == Function::Tag::CALLABLE) {
535 _scratch << "static inline ";
536 if (f.return_type() != nullptr) {
537 _emit_type_name(f.return_type());
538 } else {
539 _scratch << "void";
540 }
541 _scratch << " custom_" << hash_to_string(f.hash());
542 } else [[unlikely]] {
543 LUISA_ERROR_WITH_LOCATION("Invalid function type.");
544 }
545 _scratch << "(";
546 auto any_arg = false;
547 for (auto arg : f.arguments()) {
548 _scratch << "\n ";
549 _emit_variable_decl(arg, false);
550 _scratch << ",";
551 any_arg = true;
552 }
553 if (f.tag() == Function::Tag::KERNEL) {
554 _scratch << "\n"
555 << " const uint3 tid,\n"
556 << " const uint3 did,\n"
557 << " const uniform uint3 bid,\n"
558 << " const uniform uint3 ls) {"
559 << " if (any(binary_ge(did, ls))) { return; }\n";
560 } else {
561 if (any_arg) { _scratch.pop_back(); }
562 _scratch << ") {";
563 }
564 _indent = 1;
565 _emit_variable_declarations(f.body());
566 _indent = 0;
567 _emit_scoped_variables(f.body()->scope());
568 _emit_statements(f.body()->scope()->statements());
569 _scratch << "}\n\n";
570
571 // entry point
572 if (f.tag() == Function::Tag::KERNEL) {
573 _scratch << "struct Params {\n";
574 auto pad_count = 0u;
575 auto size = 0u;
576 for (auto arg : f.arguments()) {
577 auto aligned_size = (size + 15u) / 16u * 16u;
578 if (auto pad_size = aligned_size - size) {
579 if (pad_size % 4u == 0u) {
580 _scratch << " uint _pad_" << pad_count++
581 << "[" << pad_size / 4u << "];\n";
582 } else {
583 _scratch << " uint8 _pad_" << pad_count++
584 << "[" << pad_size << "];\n";
585 }
586 }
587 _scratch << " ";
588 _emit_variable_decl(arg, !arg.type()->is_buffer());
589 _scratch << ";\n";
590 if (arg.type()->is_buffer()) {
591 size = aligned_size + buffer_handle_size;
592 } else if (arg.type()->is_texture()) {
593 size = aligned_size + texture_handle_size;
594 } else if (arg.type()->is_accel()) {
595 size = aligned_size + accel_handle_size;
596 } else if (arg.type()->is_bindless_array()) {
597 size = aligned_size + bindless_array_handle_size;
598 } else {
599 size = aligned_size + arg.type()->size();
600 }
601 }
602 _scratch << "};\n\n";
603
604 _scratch << "export void kernel_main(\n"
605 << " const Params *uniform params,\n"

666static constexpr std::string_view float_array_3 = "array<float,3>";
667static constexpr std::string_view ray_type_desc = "struct<16,array<float,3>,float,array<float,3>,float>";
668static constexpr std::string_view hit_type_desc = "struct<16,uint,uint,vector<float,2>>";
669
670void ISPCCodegen::visit(const Type *type) noexcept {
671 if (type->is_array() &&
672 type->description() != float_array_3) {
673 _scratch << "make_array_type(";
674 _emit_type_name(type);
675 _scratch << ", ";
676 _emit_type_name(type->element());
677 _scratch << ", "
678 << type->dimension()
679 << ");\n\n";
680 } else if (type->is_structure() &&
681 type->description() != ray_type_desc &&
682 type->description() != hit_type_desc) {
683 _scratch << "struct ";
684 _emit_type_name(type);
685 _scratch << " {\n";
686 auto pad_count = 0u;
687 auto size = 0u;
688 for (auto i = 0u; i < type->members().size(); i++) {
689 auto member = type->members()[i];
690 auto a = member->alignment();
691 auto aligned_size = (size + a - 1u) / a * a;
692 if (auto pad_size = aligned_size - size) {
693 if (pad_size % 4u == 0u) {
694 _scratch << " uint _pad_" << pad_count++
695 << "[" << pad_size / 4u << "];\n";
696 } else {
697 _scratch << " uint8 _pad_" << pad_count++
698 << "[" << pad_size << "];\n";
699 }
700 }
701 _scratch << " ";
702 _emit_type_name(member);
703 _scratch << " m" << i << ";\n";
704 size = aligned_size + member->size();
705 }
706 if (auto pad_size = type->size() - size) {
707 if (pad_size % 4u == 0u) {
708 _scratch << " uint _pad_" << pad_count
709 << "[" << pad_size / 4u << "];\n";
710 } else {
711 _scratch << " uint8 _pad_" << pad_count
712 << "[" << pad_size << "];\n";
713 }
714 }
715 _scratch << "};\n\n";
716 }
717
718 // for bindless buffers
719 if (type->is_basic() || type->is_array() || type->is_structure()) {
720 _scratch << "static inline ";
721 _emit_type_name(type);
722 _scratch << " bindless_buffer_read_"
723 << luisa::format("{:016X}", type->hash())
724 << "(uniform const LCBindlessArray array, uint buffer_id, uint element_id) {\n"
725 << " const ";
726 _emit_type_name(type);
727 _scratch << " *buffer = (const ";
728 _emit_type_name(type);
729 _scratch << " *)bindless_buffer(array, buffer_id);\n";
730 _scratch << " ";
731 _emit_type_name(type);
732 _scratch << " element = buffer[element_id];\n"
733 << " return element;\n"
734 << "}\n\n";
735 }
736}

740 case Type::Tag::BOOL: _scratch << "char"; break;
741 case Type::Tag::FLOAT: _scratch << "float"; break;
