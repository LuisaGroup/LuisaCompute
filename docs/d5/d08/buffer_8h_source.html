72class BufferView {
73

76private:

87 BufferView(uint64_t handle, size_t offset_bytes, size_t size, size_t total_size) noexcept
88 : _handle{handle}, _offset_bytes{offset_bytes}, _size{size}, _total_size{total_size} {
89 if (_offset_bytes % alignof(T) != 0u) [[unlikely]] {
90 LUISA_ERROR_WITH_LOCATION(
91 "Invalid buffer view offset {} for elements with alignment {}.",
92 _offset_bytes, alignof(T));
93 }
94 }

97 BufferView(const Buffer<T> &buffer) noexcept : BufferView{buffer.view()} {}

99 [[nodiscard]] auto handle() const noexcept { return _handle; }
100 [[nodiscard]] auto size() const noexcept { return _size; }
101 [[nodiscard]] auto offset() const noexcept { return _offset_bytes / sizeof(T); }
102 [[nodiscard]] auto offset_bytes() const noexcept { return _offset_bytes; }
103 [[nodiscard]] auto size_bytes() const noexcept { return _size * sizeof(T); }
104
105 [[nodiscard]] auto original() const noexcept {
106 return BufferView{_handle, 0u, _total_size, _total_size};
107 }
108 [[nodiscard]] auto subview(size_t offset_elements, size_t size_elements) const noexcept {
109 if (size_elements + offset_elements > _size) [[unlikely]] {
110 LUISA_ERROR_WITH_LOCATION(
111 "Subview (with offset_elements = {}, size_elements = {}) "
112 "overflows buffer view (with size_elements = {}).",
113 offset_elements, size_elements, _size);
114 }
115 return BufferView{_handle, _offset_bytes + offset_elements * sizeof(T), size_elements, _total_size};
116 }
117
118 template<typename U>
119 [[nodiscard]] auto as() const noexcept {
120 if (this->size_bytes() < sizeof(U)) [[unlikely]] {
121 LUISA_ERROR_WITH_LOCATION(
122 "Unable to hold any element (with size = {}) in buffer view (with size = {}).",
123 sizeof(U), this->size_bytes());
124 }
125 return BufferView<U>{_handle, _offset_bytes, this->size_bytes() / sizeof(U), _total_size};
126 }

128 [[nodiscard]] auto copy_to(void *data) const {
129 return BufferDownloadCommand::create(_handle, offset_bytes(), size_bytes(), data);
130 }
131
132 [[nodiscard]] auto copy_from(const void *data) {

137 if (source.size() != this->size()) [[unlikely]] {
138 LUISA_ERROR_WITH_LOCATION(
139 "Incompatible buffer views with different element counts (src = {}, dst = {}).",
140 source.size(), this->size());
141 }
142 return BufferCopyCommand::create(
143 source.handle(), this->handle(),
144 source.offset_bytes(), this->offset_bytes(),
145 this->size_bytes());
146 }

225class BufferArena {
226
227private:
228 std::mutex _mutex;
229 Device &_device;
230 luisa::vector<luisa::unique_ptr<Resource>> _buffers;
231 luisa::optional<BufferView<float4>> _current_buffer;
232 size_t _capacity;
233
234public:
235 explicit BufferArena(Device &device, size_t capacity = 4_mb) noexcept
236 : _device{device}, _capacity{std::max(next_pow2(capacity), 64_kb) / sizeof(float4)} {}
237
238 template<typename T>
239 [[nodiscard]] BufferView<T> allocate(size_t n) noexcept {
240 static_assert(alignof(T) <= 16u);
241 std::scoped_lock lock{_mutex};
242 auto size = n * sizeof(T);
243 auto n_elem = (size + sizeof(float4) - 1u) / sizeof(float4);
244 if (n_elem > _capacity) {// too big, will not use the arena
245 auto buffer = luisa::make_unique<Buffer<T>>(_device.create_buffer<T>(n));
246 auto view = buffer->view();
247 _buffers.emplace_back(std::move(buffer));
248 return view;
249 }
250 if (!_current_buffer || n_elem > _current_buffer->size()) {
251 auto buffer = luisa::make_unique<Buffer<float4>>(
252 _device.create_buffer<float4>(_capacity));
253 _current_buffer = buffer->view();
254 _buffers.emplace_back(std::move(buffer));
255 }
256 auto view = _current_buffer->subview(0u, n_elem);
257 _current_buffer = _current_buffer->subview(
258 n_elem, _current_buffer->size() - n_elem);
259 return view.template as<T>().subview(0u, n);
260 }
261};
