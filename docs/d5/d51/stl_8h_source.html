52template<typename T = std::byte>
53struct allocator {
54 using value_type = T;
55 constexpr allocator() noexcept = default;
56 template<typename U>
57 constexpr allocator(allocator<U>) noexcept {}
58 [[nodiscard]] auto allocate(std::size_t n) const noexcept {
59 return static_cast<T *>(detail::allocator_allocate(sizeof(T) * n, alignof(T)));
60 }
61 void deallocate(T *p, size_t) const noexcept {
62 detail::allocator_deallocate(p, alignof(T));
63 }
64 template<typename R>
65 [[nodiscard]] constexpr auto operator==(allocator<R>) const noexcept -> bool {
66 return std::is_same_v<T, R>;
67 }
68};

70template<typename T>
71[[nodiscard]] inline auto allocate(size_t n = 1u) noexcept {
72 return allocator<T>{}.allocate(n);
73}

75template<typename T>
76inline void deallocate(T *p) noexcept {
77 allocator<T>{}.deallocate(p, 0u);

80template<typename T, typename... Args>
81[[nodiscard]] inline auto new_with_allocator(Args &&...args) noexcept {

85template<typename T>
86inline void delete_with_allocator(T *p) noexcept {
