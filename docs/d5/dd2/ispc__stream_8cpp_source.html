18void check_texture_boundary(ISPCTexture *tex, uint level, uint3 size) {
19 // debug: check boundary
20 if (level >= tex->lodLevel)
21 LUISA_ERROR_WITH_LOCATION("check texture boundary: lod={} out of bound", level);
22 if (size.x > max(tex->size[0] >> level, 1u))
23 LUISA_ERROR_WITH_LOCATION("check texture boundary: out of bound");
24 if (size.y > max(tex->size[1] >> level, 1u))
25 LUISA_ERROR_WITH_LOCATION("check texture boundary: out of bound");
26 if (size.z > max(tex->size[2] >> level, 1u))
27 LUISA_ERROR_WITH_LOCATION("check texture boundary: out of bound");
28};

30void ISPCStream::dispatch(const CommandList &cmd_list) noexcept {
31 for (auto cmd : cmd_list) {
32 while (_pool.task_count() > _pool.size() * 4u) {
33 using namespace std::chrono_literals;
34 std::this_thread::sleep_for(1ms);
35 }
36 cmd->accept(*this);
37 }
38 _pool.barrier();
39}

75void ISPCStream::visit(const BufferToTextureCopyCommand *command) noexcept {
76 _pool.async([cmd = *command] {
77 ISPCTexture *tex = reinterpret_cast<ISPCTexture *>(cmd.texture());
78 check_texture_boundary(tex, cmd.level(), cmd.size());
79 // calc stride
80 uint pxsize = pixel_storage_size(tex->storage);
81 uint texsizex = max(tex->size[0] >> cmd.level(), 1u);
82 uint texsizey = max(tex->size[1] >> cmd.level(), 1u);
83 int tex_r_stride = texsizex * pxsize;
84 int tex_p_stride = texsizex * texsizey * pxsize;
85 int target_r_stride = cmd.size().x * pxsize;
86 int target_p_stride = cmd.size().x * cmd.size().y * pxsize;
87 // copy data
88 for (int dz = 0; dz < cmd.size().z; ++dz)
89 for (int dy = 0; dy < cmd.size().y; ++dy)
90 memcpy((unsigned char *)tex->lods[cmd.level()] + dz * tex_p_stride + dy * tex_r_stride,
91 (unsigned char *)cmd.buffer() + cmd.buffer_offset() + dz * target_p_stride + dy * target_r_stride,
92 cmd.size().x * pxsize);
93 });
94}

96void ISPCStream::visit(const ShaderDispatchCommand *command) noexcept {
97 auto shader = reinterpret_cast<const ISPCShader *>(command->handle());
98 luisa::vector<std::byte> argument_buffer(shader->argument_buffer_size());
99 command->decode([&](auto vid, auto argument) noexcept {
100 auto ptr = argument_buffer.data() + shader->argument_offset(vid);
101 using T = decltype(argument);
102 if constexpr (std::is_same_v<T, ShaderDispatchCommand::BufferArgument>) {
103 auto buffer = reinterpret_cast<void *>(argument.handle + argument.offset);
104 std::memcpy(ptr, &buffer, sizeof(buffer));
105 } else if constexpr (std::is_same_v<T, ShaderDispatchCommand::TextureArgument>) {
106 auto texture = reinterpret_cast<const ISPCTexture *>(argument.handle);
107 auto handle = ISPCTexture::TextureView{texture->handle().ptr, argument.level, 0};
108 std::memcpy(ptr, &handle, sizeof(handle));
109 } else if constexpr (std::is_same_v<T, ShaderDispatchCommand::BindlessArrayArgument>) {
110 auto array = reinterpret_cast<const ISPCBindlessArray *>(argument.handle);
111 auto handle = array->handle();
112 std::memcpy(ptr, &handle, sizeof(handle));
113 } else if constexpr (std::is_same_v<T, ShaderDispatchCommand::AccelArgument>) {
114 auto accel = reinterpret_cast<const ISPCAccel *>(argument.handle);
115 auto handle = accel->handle();
116 std::memcpy(ptr, &handle, sizeof(handle));
117 } else {// uniform
118 static_assert(std::same_as<T, luisa::span<const std::byte>>);
119 std::memcpy(ptr, argument.data(), argument.size_bytes());
120 }
121 });
122 auto shared_buffer = luisa::make_shared<luisa::vector<std::byte>>(std::move(argument_buffer));
123 auto dispatch_size = command->dispatch_size();
124 auto block_size = command->kernel().block_size();
125 auto grid_size = (dispatch_size + block_size - 1u) / block_size;
126 // LUISA_INFO(
127 // "Dispatching ISPC kernel "
128 // "with ({}, {}, {}) blocks, "
129 // "each with ({}, {}, {}) threads.",
130 // grid_size.x, grid_size.y, grid_size.z,
131 // block_size.x, block_size.y, block_size.z);
132 _pool.parallel(
133 grid_size.x, grid_size.y, grid_size.z,
134 [shared_buffer, dispatch_size, module = shader->shared_module()](auto bx, auto by, auto bz) noexcept {
135 module->invoke(shared_buffer->data(), luisa::make_uint3(bx, by, bz), dispatch_size);
136 });
137}

181void ISPCStream::visit(const TextureCopyCommand *command) noexcept {
182 _pool.async([cmd = *command] {
183 ISPCTexture *src_tex = reinterpret_cast<ISPCTexture *>(cmd.src_handle());
184 ISPCTexture *dst_tex = reinterpret_cast<ISPCTexture *>(cmd.dst_handle());
185 check_texture_boundary(src_tex, cmd.src_level(), cmd.size());
186 check_texture_boundary(dst_tex, cmd.dst_level(), cmd.size());
187 // calc stride
188 uint pxsize = pixel_storage_size(src_tex->storage);
189 uint src_sizex = max(src_tex->size[0] >> cmd.src_level(), 1u);
190 uint dst_sizex = max(dst_tex->size[0] >> cmd.dst_level(), 1u);
191 uint src_sizey = max(src_tex->size[1] >> cmd.src_level(), 1u);
192 uint dst_sizey = max(dst_tex->size[1] >> cmd.dst_level(), 1u);
193 int src_r_stride = src_sizex * pxsize;
194 int src_p_stride = src_sizex * src_sizey * pxsize;
195 int dst_r_stride = dst_sizex * pxsize;
196 int dst_p_stride = dst_sizex * dst_sizey * pxsize;
197 // copy data
198 for (int dz = 0; dz < cmd.size().z; ++dz)
199 for (int dy = 0; dy < cmd.size().y; ++dy)
200 memcpy((unsigned char *)dst_tex->lods[cmd.dst_level()] + dz * dst_p_stride + dy * dst_r_stride,
201 (unsigned char *)src_tex->lods[cmd.src_level()] + dz * src_p_stride + dy * src_r_stride,
