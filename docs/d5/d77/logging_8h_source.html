38 auto error_message = fmt::format(std::forward<Args>(args)...);
39 auto trace = luisa::backtrace();
40 for (auto i = 0u; i < trace.size(); i++) {
41 auto &&t = trace[i];
42 using namespace std::string_view_literals;
43 error_message.append(fmt::format(
44 FMT_STRING("\n {:>2} [0x{:012x}]: {} :: {} + {}"sv),
45 i, t.address, t.module, t.symbol, t.offset));
46 }
47 detail::default_logger().error("{}", error_message);
48 std::abort();
49}
