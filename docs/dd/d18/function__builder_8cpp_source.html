21void FunctionBuilder::pop(FunctionBuilder *func) noexcept {
22 if (_function_stack().empty()) [[unlikely]] {
23 LUISA_ERROR_WITH_LOCATION("Invalid pop on empty function stack.");
24 }
25 auto f = _function_stack().back();
26 _function_stack().pop_back();
27 if (func != nullptr && f != func) [[unlikely]] {
28 LUISA_ERROR_WITH_LOCATION("Invalid function on stack top.");
29 }
30 if (f->tag() == Function::Tag::CALLABLE &&
31 !f->builtin_variables().empty()) [[unlikely]] {
32 LUISA_ERROR_WITH_LOCATION(
33 "Custom callables may not use builtin "
34 "variables. Pass them as arguments "
35 "if you want to use their values.");
36 }
37 if (f->_raytracing &&
38 (f->_using_shared_storage ||
39 f->_used_builtin_callables.test(
40 CallOp::SYNCHRONIZE_BLOCK))) [[unlikely]] {
41 LUISA_ERROR_WITH_LOCATION(
42 "Raytracing functions are not allowed to "
43 "use shared storage or call synchronize_block().");
44 }
45 if (f->_arguments.size() != f->_argument_bindings.size()) {
46 LUISA_ERROR_WITH_LOCATION(
47 "Arguments and their bindings have different sizes ({} and {}).",
48 f->_arguments.size(), f->_argument_bindings.size());
49 }
50
51 // hash
52 f->_compute_hash();
53
54 // clear temporary data
55 for (auto p : f->_temporary_data) {
56 luisa::detail::allocator_deallocate(
57 p.first, p.second);
58 }
59 f->_temporary_data.clear();
60 f->_temporary_data.shrink_to_fit();
61}

412const CallExpr *FunctionBuilder::call(const Type *type, Function custom, luisa::span<const Expression *const> args) noexcept {
413 if (custom.tag() != Function::Tag::CALLABLE) {
414 LUISA_ERROR_WITH_LOCATION(
415 "Calling non-callable function in device code.");
416 }
417 auto f = custom.builder();
418 if (f->raytracing()) { _raytracing = true; }
419 CallExpr::ArgumentList call_args(f->_arguments.size(), nullptr);
420 auto in_iter = args.begin();
421 for (auto i = 0u; i < f->_arguments.size(); i++) {
422 call_args[i] = luisa::visit(
423 [&]<typename T>(T binding) noexcept -> const Expression * {
424 if constexpr (std::is_same_v<T, BufferBinding>) {
425 return buffer_binding(f->_arguments[i].type(), binding.handle, binding.offset_bytes);
426 } else if constexpr (std::is_same_v<T, TextureBinding>) {
427 return texture_binding(f->_arguments[i].type(), binding.handle, binding.level);
428 } else if constexpr (std::is_same_v<T, BindlessArrayBinding>) {
429 return bindless_array_binding(binding.handle);
430 } else if constexpr (std::is_same_v<T, AccelBinding>) {
431 return accel_binding(binding.handle);
432 } else {
433 return *(in_iter++);
434 }
435 },
436 f->_argument_bindings[i]);
437 }
438 if (in_iter != args.end()) [[unlikely]] {
439 LUISA_ERROR_WITH_LOCATION(
440 "Invalid call arguments for custom callable #{}.",
441 custom.hash());
442 }
443 auto expr = _create_expression<CallExpr>(type, custom, std::move(call_args));
444 if (auto iter = std::find_if(
445 _used_custom_callables.cbegin(),
446 _used_custom_callables.cend(),
447 [c = custom.builder()](auto &&p) noexcept { return c == p.get(); });
448 iter == _used_custom_callables.cend()) {
449 _used_custom_callables.emplace_back(custom.shared_builder());
450 }
451 return expr;
452}
