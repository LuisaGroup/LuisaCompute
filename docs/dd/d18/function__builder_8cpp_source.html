21void FunctionBuilder::pop(FunctionBuilder *func) noexcept {
22 if (_function_stack().empty()) [[unlikely]] {
23 LUISA_ERROR_WITH_LOCATION("Invalid pop on empty function stack.");
24 }
25 auto f = _function_stack().back();
26 _function_stack().pop_back();
27 if (func != nullptr && f != func) [[unlikely]] {
28 LUISA_ERROR_WITH_LOCATION("Invalid function on stack top.");
29 }
30 if (f->tag() == Function::Tag::CALLABLE &&
31 !f->builtin_variables().empty()) [[unlikely]] {
32 LUISA_ERROR_WITH_LOCATION(
33 "Custom callables may not use builtin "
34 "variables. Pass them as arguments "
35 "if you want to use their values.");
36 }
37 if (f->_raytracing &&
38 (f->_using_shared_storage ||
39 f->_used_builtin_callables.test(
40 CallOp::SYNCHRONIZE_BLOCK))) [[unlikely]] {
41 LUISA_ERROR_WITH_LOCATION(
42 "Raytracing functions are not allowed to "
43 "use shared storage or call synchronize_block().");
44 }
45 if (f->_arguments.size() != f->_argument_bindings.size()) {
46 LUISA_ERROR_WITH_LOCATION(
47 "Arguments and their bindings have different sizes ({} and {}).",
48 f->_arguments.size(), f->_argument_bindings.size());
49 }
50
51 // hash
52 f->_compute_hash();
53
54 // clear temporary data
55 for (auto p : f->_temporary_data) {
56 luisa::detail::allocator_deallocate(
57 p.first, p.second);
58 }
59 f->_temporary_data.clear();
60 f->_temporary_data.shrink_to_fit();
61}

426const CallExpr *FunctionBuilder::call(const Type *type, Function custom, luisa::span<const Expression *const> args) noexcept {
427 if (custom.tag() != Function::Tag::CALLABLE) {
428 LUISA_ERROR_WITH_LOCATION(
429 "Calling non-callable function in device code.");
430 }
431 auto f = custom.builder();
432 if (f->raytracing()) { _raytracing = true; }
433 CallExpr::ArgumentList call_args(f->_arguments.size(), nullptr);
434 auto in_iter = args.begin();
435 for (auto i = 0u; i < f->_arguments.size(); i++) {
436 call_args[i] = luisa::visit(
437 [&]<typename T>(T binding) noexcept -> const Expression * {
438 if constexpr (std::is_same_v<T, BufferBinding>) {
439 return buffer_binding(f->_arguments[i].type(), binding.handle, binding.offset_bytes);
440 } else if constexpr (std::is_same_v<T, TextureBinding>) {
441 return texture_binding(f->_arguments[i].type(), binding.handle, binding.level);
442 } else if constexpr (std::is_same_v<T, BindlessArrayBinding>) {
443 return bindless_array_binding(binding.handle);
444 } else if constexpr (std::is_same_v<T, AccelBinding>) {
445 return accel_binding(binding.handle);
446 } else {
447 return *(in_iter++);
448 }
449 },
450 f->_argument_bindings[i]);
451 }
452 if (in_iter != args.end()) [[unlikely]] {
453 LUISA_ERROR_WITH_LOCATION(
454 "Invalid call arguments for custom callable #{}.",
455 custom.hash());
456 }
457 auto expr = _create_expression<CallExpr>(type, custom, std::move(call_args));
458 if (auto iter = std::find_if(
459 _used_custom_callables.cbegin(),
460 _used_custom_callables.cend(),
461 [c = custom.builder()](auto &&p) noexcept { return c == p.get(); });
462 iter == _used_custom_callables.cend()) {
463 _used_custom_callables.emplace_back(custom.shared_builder());
464 }
465 return expr;
466}
