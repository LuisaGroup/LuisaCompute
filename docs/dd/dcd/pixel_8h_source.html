179template<typename T>
180[[nodiscard]] constexpr auto pixel_storage_to_format(PixelStorage storage) noexcept {
181 if constexpr (std::is_same_v<T, float>) {
182 switch (storage) {
183 case PixelStorage::BYTE1: return PixelFormat::R8UNorm;
184 case PixelStorage::BYTE2: return PixelFormat::RG8UNorm;
185 case PixelStorage::BYTE4: return PixelFormat::RGBA8UNorm;
186 case PixelStorage::SHORT1: return PixelFormat::R16UNorm;
187 case PixelStorage::SHORT2: return PixelFormat::RG16UNorm;
188 case PixelStorage::SHORT4: return PixelFormat::RGBA16UNorm;
189 case PixelStorage::HALF1: return PixelFormat::R16F;
190 case PixelStorage::HALF2: return PixelFormat::RG16F;
191 case PixelStorage::HALF4: return PixelFormat::RGBA16F;
192 case PixelStorage::FLOAT1: return PixelFormat::R32F;
193 case PixelStorage::FLOAT2: return PixelFormat::RG32F;
194 case PixelStorage::FLOAT4: return PixelFormat::RGBA32F;
195 default: LUISA_ERROR_WITH_LOCATION("Invalid pixel storage for float format.");
196 }
197 } else if constexpr (std::is_same_v<T, int>) {
198 switch (storage) {
199 case PixelStorage::BYTE1: return PixelFormat::R8SInt;
200 case PixelStorage::BYTE2: return PixelFormat::RG8SInt;
201 case PixelStorage::BYTE4: return PixelFormat::RGBA8SInt;
202 case PixelStorage::SHORT1: return PixelFormat::R16SInt;
203 case PixelStorage::SHORT2: return PixelFormat::RG16SInt;
204 case PixelStorage::SHORT4: return PixelFormat::RGBA16SInt;
205 case PixelStorage::INT1: return PixelFormat::R32SInt;
206 case PixelStorage::INT2: return PixelFormat::RG32SInt;
207 case PixelStorage::INT4: return PixelFormat::RGBA32SInt;
208 default: LUISA_ERROR_WITH_LOCATION("Invalid pixel storage for int format.");
209 }
210 } else if constexpr (std::is_same_v<T, uint>) {
211 switch (storage) {
212 case PixelStorage::BYTE1: return PixelFormat::R8UInt;
213 case PixelStorage::BYTE2: return PixelFormat::RG8UInt;
214 case PixelStorage::BYTE4: return PixelFormat::RGBA8UInt;
215 case PixelStorage::SHORT1: return PixelFormat::R16UInt;
216 case PixelStorage::SHORT2: return PixelFormat::RG16UInt;
217 case PixelStorage::SHORT4: return PixelFormat::RGBA16UInt;
218 case PixelStorage::INT1: return PixelFormat::R32UInt;
219 case PixelStorage::INT2: return PixelFormat::RG32UInt;
220 case PixelStorage::INT4: return PixelFormat::RGBA32UInt;
221 default: LUISA_ERROR_WITH_LOCATION("Invalid pixel storage for uint format.");
222 }
223 } else {
224 static_assert(always_false_v<T>);
225 }
226 return PixelFormat{};
227}
