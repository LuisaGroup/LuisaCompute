33template<typename T>
34[[nodiscard]] inline auto extract_expression(T &&v) noexcept {
35 if constexpr (is_dsl_v<T>) {
36 return std::forward<T>(v).expression();
37 } else {
38 static_assert(concepts::basic<T>);
39 return FunctionBuilder::current()->literal(Type::of<T>(), std::forward<T>(v));

226template<typename T>
227struct ExprEnableStaticCast {
228 template<typename Dest>
229 requires concepts::static_convertible<expr_value_t<T>, expr_value_t<Dest>>
230 [[nodiscard]] auto cast() const noexcept {
231 auto src = def(*static_cast<const T *>(this));
232 using TrueDest = expr_value_t<Dest>;
233 return def<TrueDest>(
234 FunctionBuilder::current()->cast(
235 Type::of<TrueDest>(),
236 CastOp::STATIC,
237 src.expression()));
238 }
239};
240
242template<typename T>
243struct ExprEnableBitwiseCast {
244 template<typename Dest>
245 requires concepts::bitwise_convertible<expr_value_t<T>, expr_value_t<Dest>>
246 [[nodiscard]] auto as() const noexcept {
247 auto src = def(*static_cast<const T *>(this));
248 using TrueDest = expr_value_t<Dest>;
249 return def<TrueDest>(
250 FunctionBuilder::current()->cast(
251 Type::of<TrueDest>(),
252 CastOp::BITWISE,
253 src.expression()));
254 }
255};

258template<typename T>
259struct ExprEnableSubscriptAccess {
260 template<typename I>
261 requires is_integral_expr_v<I>
262 [[nodiscard]] auto operator[](I &&index) const noexcept {
263 auto self = def<T>(static_cast<const T *>(this)->expression());
264 using Elem = std::remove_cvref_t<
265 decltype(std::declval<expr_value_t<T>>()[0])>;
266 return def<Elem>(FunctionBuilder::current()->access(
267 Type::of<Elem>(), self.expression(),
268 extract_expression(std::forward<I>(index))));
269 }
270};
271
273template<typename T>
274struct ExprEnableGetMemberByIndex {
275 template<size_t i>
276 [[nodiscard]] auto get() const noexcept {
277 static_assert(i < dimension_v<expr_value_t<T>>);
278 return static_cast<const T *>(this)->operator[](static_cast<uint>(i));
279 }
280};
