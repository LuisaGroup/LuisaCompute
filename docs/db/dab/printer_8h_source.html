78template<typename... Args>
79void Printer::log(Args &&...args) noexcept {
80 auto count = 1u /* desc_id */ + static_cast<uint>(sizeof...(args));
81 auto size = static_cast<uint>(_buffer.size() - 1u);
82 auto offset = _buffer.atomic(size).fetch_add(count);
83 auto process_arg = [&]<typename Arg>(Arg &&arg) noexcept {
84 offset = offset + 1u;
85 using T = expr_value_t<Arg>;
86 if constexpr (std::is_same_v<T, bool>) {
87 _buffer.write(offset, cast<uint>(std::forward<Arg>(arg)));
88 return Descriptor::Tag::BOOL;
89 } else if constexpr (std::is_same_v<T, int>) {
90 _buffer.write(offset, cast<uint>(std::forward<Arg>(arg)));
91 return Descriptor::Tag::INT;
92 } else if constexpr (std::is_same_v<T, uint>) {
93 _buffer.write(offset, cast<uint>(std::forward<Arg>(arg)));
94 return Descriptor::Tag::UINT;
95 } else if constexpr (std::is_same_v<T, float>) {
96 _buffer.write(offset, as<uint>(std::forward<Arg>(arg)));
97 return Descriptor::Tag::FLOAT;
98 } else {
99 luisa::string s{std::forward<Arg>(arg)};
100 auto [iter, not_present] = _string_id.try_emplace(
101 std::move(s), static_cast<uint>(_strings.size()));
102 if (not_present) { _strings.emplace_back(iter->first); }
103 _buffer.write(offset, iter->second);
104 return Descriptor::Tag::STRING;
105 }
106 };
107 auto index = offset;
108 Descriptor desc;
109 if_(offset + count <= size, [&] {
110 desc.value_tags = {
111 process_arg(std::forward<Args>(args))...};
112 });
113 if_(index < size, [&] {
114 auto [iter, not_present] = _desc_id.try_emplace(
115 std::move(desc), static_cast<uint>(_descriptors.size()));
116 if (not_present) { _descriptors.emplace_back(iter->first.value_tags); }
117 _buffer.write(index, iter->second);
118 });
119}
