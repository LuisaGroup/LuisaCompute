118 // arguments
119 _argument_offsets.reserve(func.arguments().size());
120 for (auto &&arg : func.arguments()) {
121 auto aligned_offset = (_argument_buffer_size + 15u) / 16u * 16u;
122 _argument_offsets.emplace(arg.uid(), aligned_offset);
123 if (arg.type()->is_buffer()) {
124 _argument_buffer_size = aligned_offset + ISPCCodegen::buffer_handle_size;
125 } else if (arg.type()->is_texture()) {
126 _argument_buffer_size = aligned_offset + ISPCCodegen::texture_handle_size;
127 } else if (arg.type()->is_accel()) {
128 _argument_buffer_size = aligned_offset + ISPCCodegen::accel_handle_size;
129 } else if (arg.type()->is_bindless_array()) {
130 _argument_buffer_size = aligned_offset + ISPCCodegen::bindless_array_handle_size;
131 } else {
132 _argument_buffer_size = aligned_offset + arg.type()->size();
133 }
134 }
135 _argument_buffer_size = (_argument_buffer_size + 15u) / 16u * 16u;

138size_t ISPCShader::argument_offset(uint uid) const noexcept {
139 if (auto iter = _argument_offsets.find(uid);

142 }
143 LUISA_ERROR_WITH_LOCATION("Invalid argument uid {}.", uid);
