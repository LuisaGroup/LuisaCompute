49private:

52public:
54 explicit constexpr Hash64(uint64_t seed = default_seed) noexcept
55 : _seed{seed} {}

58 template<typename T>
59 [[nodiscard]] auto operator()(T &&s) const noexcept -> uint64_t {
60 if constexpr (detail::hashable_with_hash_method<T>) {
61 return (*this)(std::forward<T>(s).hash());
62 } else if constexpr (detail::hashable_with_hash_code_method<T>) {
63 return (*this)(std::forward<T>(s).hash_code());
64 } else if constexpr (concepts::string_viewable<T>) {
65 std::string_view sv{std::forward<T>(s)};
66 return detail::xxh3_hash64(sv.data(), sv.size(), _seed);
67 } else if constexpr (is_vector3_v<T>) {
68 auto x = s;
69 return detail::xxh3_hash64(&x, sizeof(vector_element_t<T>) * 3u, _seed);
70 } else if constexpr (is_matrix3_v<T>) {
71 auto x = luisa::make_float4x4(s);
72 return (*this)(x);
73 } else if constexpr (
74 std::is_arithmetic_v<std::remove_cvref_t<T>> ||
75 std::is_enum_v<std::remove_cvref_t<T>> ||
76 is_basic_v<T>) {
77 auto x = s;
78 return detail::xxh3_hash64(&x, sizeof(x), _seed);
79 } else {
80 static_assert(always_false_v<T>);
81 }
82 }
83};
