26[[nodiscard]] auto runtime_directory(const std::filesystem::path &p) noexcept {
27 auto cp = std::filesystem::canonical(p);

58Device Context::create_device(std::string_view backend_name, const nlohmann::json &properties) noexcept {
59 if (!properties.is_object()) {
60 LUISA_ERROR_WITH_LOCATION(
61 "Invalid device properties: {}.",
62 properties.dump());
63 }
64 auto [create, destroy] = [backend_name, this] {
65 if (auto iter = std::find(_impl->device_identifiers.cbegin(),
66 _impl->device_identifiers.cend(),
67 backend_name);
68 iter != _impl->device_identifiers.cend()) {
69 auto i = iter - _impl->device_identifiers.cbegin();
70 auto c = _impl->device_creators[i];
71 auto d = _impl->device_deleters[i];
72 return std::make_pair(c, d);
73 }
74 auto &&m = _impl->loaded_modules.emplace_back(
75 _impl->runtime_directory / "backends",
76 fmt::format("luisa-compute-backend-{}", backend_name));
77 auto c = m.function<Device::Creator>("create");
78 auto d = m.function<Device::Deleter>("destroy");
79 _impl->device_identifiers.emplace_back(backend_name);
80 _impl->device_creators.emplace_back(c);
81 _impl->device_deleters.emplace_back(d);
82 return std::make_pair(c, d);
83 }();
84 return Device{Device::Handle{create(*this, properties.dump()), destroy}};
85}
