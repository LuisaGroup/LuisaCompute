13luisa::shared_ptr<ISPCModule> ISPCDLLModule::load(
14 const Context &ctx, const std::filesystem::path &obj_path) noexcept {
15
16#ifdef LUISA_PLATFORM_WINDOWS
17 auto &&support_dir = ctx.runtime_directory();
18 auto link_exe = support_dir / "link.exe";
19 auto crt_path = support_dir / "msvcrt.lib";
20 auto embree_path = support_dir / "embree3.lib";
21 auto dll_path = obj_path;
22 auto lib_path = obj_path;
23 auto exp_path = obj_path;
24 dll_path.replace_extension("dll");
25 lib_path.replace_extension("lib");
26 exp_path.replace_extension("exp");
27
28 auto command = luisa::format(
29 R"({} /DLL /NOLOGO /OUT:"{}" /DYNAMICBASE "{}" "{}" /DEBUG:NONE /NOENTRY /EXPORT:kernel_main /NODEFAULTLIB "{}")",
30 link_exe.string(), dll_path.string(), crt_path.string(), embree_path.string(), obj_path.string());
31 LUISA_INFO("Generating DLL for ISPC kernel: {}", command);
32 if (auto exit_code = system(command.c_str()); exit_code != 0) {
33 LUISA_ERROR_WITH_LOCATION(
34 "Failed to generate DLL for IPSC kernel. "
35 "Linker exit with 0x{:02x}.",
36 exit_code);
37 }
38 std::filesystem::remove(lib_path);
39 std::filesystem::remove(exp_path);
40 return luisa::make_unique<ISPCDLLModule>(ISPCDLLModule{
41 DynamicModule{dll_path.parent_path(), dll_path.stem().string()}});
42#else
43
44 auto embree_name = [&ctx] {
45 std::filesystem::directory_iterator runtime_files{ctx.runtime_directory()};
46 for (auto &&file : runtime_files) {
47 auto filename = file.path().filename();
48 auto stem = filename.stem().string();
49 auto ext = filename.extension().string();
50 if (stem.find("embree3") != std::string::npos &&
51 (ext == ".dll" || ext == ".so" || ext == ".dylib")) {
52 return stem.starts_with("lib") ? stem.substr(3u) : stem;
53 }
54 }
55 LUISA_WARNING_WITH_LOCATION("Failed to find embree3.");
56 return std::string{};
57 }();
58
59 auto support_dir = ctx.runtime_directory() / "backends" / "ispc_support";
60 auto file_name = obj_path.filename().stem().string();
61 auto output_folder = obj_path.parent_path();
62 auto dll_path = output_folder / luisa::format("lib{}.so", file_name);
63
64#ifndef NDEBUG
65 auto link_opt = "-shared -g -O3";
66#else
67 auto link_opt = "-shared -O3";
68#endif
69
70 auto command = luisa::format(
71 R"(cc {} -o "{}" "{}")",
72 link_opt, dll_path.string(), obj_path.string());
73 if (!embree_name.empty()) {
74 command.append(luisa::format(
75 R"( -L"{}" -l{})",
76 ctx.runtime_directory().string(),
77 embree_name));
78 }
79 LUISA_INFO("Generating DLL for ISPC kernel: {}", command);
80 if (auto exit_code = system(command.c_str()); exit_code != 0) {
81 LUISA_ERROR_WITH_LOCATION(
82 "Failed to generate DLL for IPSC kernel. "
83 "Linker exit with 0x{:02x}.",
84 exit_code);
85 }
86 return luisa::make_shared<ISPCDLLModule>(ISPCDLLModule{
87 DynamicModule{output_folder, file_name}});
88#endif
89}
