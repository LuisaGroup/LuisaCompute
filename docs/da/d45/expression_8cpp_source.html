17void CallExpr::_mark() const noexcept {
18 if (is_builtin()) {
19 if (_op == CallOp::BUFFER_WRITE ||
20 _op == CallOp::TEXTURE_WRITE ||
21 _op == CallOp::ATOMIC_EXCHANGE ||
22 _op == CallOp::ATOMIC_COMPARE_EXCHANGE ||
23 _op == CallOp::ATOMIC_FETCH_ADD ||
24 _op == CallOp::ATOMIC_FETCH_SUB ||
25 _op == CallOp::ATOMIC_FETCH_AND ||
26 _op == CallOp::ATOMIC_FETCH_OR ||
27 _op == CallOp::ATOMIC_FETCH_XOR ||
28 _op == CallOp::ATOMIC_FETCH_MIN ||
29 _op == CallOp::ATOMIC_FETCH_MAX) {
30 _arguments[0]->mark(Usage::WRITE);
31 for (auto i = 1u; i < _arguments.size(); i++) {
32 _arguments[i]->mark(Usage::READ);
33 }
34 } else {
35 for (auto arg : _arguments) {
36 arg->mark(Usage::READ);
37 }
38 }
39 } else {
40 auto args = _custom.arguments();
41 for (auto i = 0u; i < args.size(); i++) {
42 auto arg = args[i];
43 _arguments[i]->mark(
44 arg.tag() == Variable::Tag::REFERENCE ||
45 arg.tag() == Variable::Tag::BUFFER ||
46 arg.tag() == Variable::Tag::TEXTURE ?
47 _custom.variable_usage(arg.uid()) :
48 Usage::READ);
49 }
50 }
51}
