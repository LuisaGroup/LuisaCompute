77template<typename T, size_t N>
78struct Vector : public detail::VectorStorage<T, N> {
79 static constexpr auto dimension = N;
80 using value_type = T;
81 using Storage = detail::VectorStorage<T, N>;
82 static_assert(std::disjunction_v<
83 std::is_same<T, bool>,
84 std::is_same<T, float>,
85 std::is_same<T, int>,
86 std::is_same<T, uint>>,
87 "Invalid vector type");
88 static_assert(N == 2 || N == 3 || N == 4, "Invalid vector dimension");
89 using Storage::VectorStorage;
90 [[nodiscard]] constexpr T &operator[](size_t index) noexcept { return (&(this->x))[index]; }
91 [[nodiscard]] constexpr const T &operator[](size_t index) const noexcept { return (&(this->x))[index]; }
92};
