24luisa::string_view Printer::retrieve(Stream &stream) noexcept {
25 if (!_reset_called) [[unlikely]] {
26 LUISA_ERROR_WITH_LOCATION(
27 "Printer results cannot be "
28 "retrieved if never reset.");
29 }
30 auto zero = 0u;
31 stream << _buffer.copy_to(_host_buffer.data())
32 << _buffer.view(_buffer.size() - 1u, 1u).copy_from(&zero)
33 << synchronize();
34 auto size = std::min(
35 static_cast<uint>(_buffer.size() - 1u),
36 _host_buffer.back());
37 _scratch.clear();
38 auto records = luisa::span{_host_buffer}.subspan(0u, size);
39 for (auto offset = 0u; offset < size;) {
40 auto desc_id = records[offset++];
41 auto desc = _descriptors[desc_id];
42 if (offset + desc.size() > records.size()) {
43 break;
44 }
45 for (auto &&tag : desc) {
46 auto record = records[offset++];
47 switch (tag) {
48 case Descriptor::Tag::INT:
49 _scratch.append(luisa::format(
50 "{}", static_cast<int>(record)));
51 break;
52 case Descriptor::Tag::UINT:
53 _scratch.append(luisa::format(
54 "{}", record));
55 break;
56 case Descriptor::Tag::FLOAT:
57 _scratch.append(luisa::format(
58 "{}", luisa::bit_cast<float>(record)));
59 break;
60 case Descriptor::Tag::BOOL:
61 _scratch.append(luisa::format(
62 "{}", static_cast<bool>(record)));
63 break;
64 case Descriptor::Tag::STRING:
65 _scratch.append(_strings[record]);
66 break;
67 }
68 }
69 _scratch.append("\n");
70 }
71 return _scratch;
72}
