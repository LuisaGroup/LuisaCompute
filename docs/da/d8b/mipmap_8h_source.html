143[[nodiscard]] constexpr auto max_mip_levels(uint3 size, uint requested_levels) noexcept {
144 auto max_size = std::max({size.x, size.y, size.z});
145 auto max_levels = 0u;
146 while (max_size != 0u) {
147 max_size >>= 1u;
148 max_levels++;
149 }
150 return requested_levels == 0u ? max_levels : std::min(requested_levels, max_levels);
151}
