84 [[nodiscard]] auto copy_from(MipmapView src) const noexcept {
85 if (!all(_size == src._size)) [[unlikely]] {
86 LUISA_ERROR_WITH_LOCATION(
87 "MipmapView sizes mismatch in copy command "
88 "(src: [{}, {}], dest: [{}, {}]).",
89 src._size.x, src._size.y, _size.x, _size.y);
90 }
91 if (src._storage != _storage) [[unlikely]] {
92 LUISA_ERROR_WITH_LOCATION(
93 "MipmapView storages mismatch "
94 "(src = {}, dst = {})",
95 to_underlying(src._storage),
96 to_underlying(_storage));
97 }
98 return TextureCopyCommand::create(
99 _storage, src._handle, _handle, src._level, _level, _size);
100 }

138[[nodiscard]] constexpr auto max_mip_levels(uint3 size, uint requested_levels) noexcept {
139 auto max_size = std::max({size.x, size.y, size.z});
140 auto max_levels = 0u;
141 while (max_size != 0u) {
142 max_size >>= 1u;
143 max_levels++;
144 }
145 return requested_levels == 0u ? max_levels : std::min(requested_levels, max_levels);
146}
