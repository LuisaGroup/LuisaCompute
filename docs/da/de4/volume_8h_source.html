20// Volumes are 3D textures without sampling, i.e., 3D surfaces.
21template<typename T>
22class Volume final : public Resource {
23
24 static_assert(std::disjunction_v<
25 std::is_same<T, int>,
26 std::is_same<T, uint>,
27 std::is_same<T, float>>);
28
29private:
30 PixelStorage _storage{};
31 uint _mip_levels{};
32 uint3 _size{};
33
34private:
35 friend class Device;
36 Volume(Device::Interface *device, PixelStorage storage, uint3 size, uint mip_levels = 1u, Sampler sampler = {}) noexcept
37 : Resource{
38 device, Tag::TEXTURE,
39 device->create_texture(
40 pixel_storage_to_format<T>(storage), 3u,
41 size.x, size.y, size.z,
42 detail::max_mip_levels(size, mip_levels))},
43 _storage{storage}, _mip_levels{detail::max_mip_levels(size, mip_levels)}, _size{size} {}
44
45public:
46 Volume() noexcept = default;
47 using Resource::operator bool;
48
49 [[nodiscard]] auto native_handle() const noexcept { return device()->texture_native_handle(handle()); }
50 [[nodiscard]] auto mip_levels() const noexcept { return _mip_levels; }
51 [[nodiscard]] auto size() const noexcept { return _size; }
52 [[nodiscard]] auto storage() const noexcept { return _storage; }
53 [[nodiscard]] auto format() const noexcept { return pixel_storage_to_format<T>(_storage); }
54
55 [[nodiscard]] auto view(uint32_t level) const noexcept {
56 if (level >= _mip_levels) [[unlikely]] {
57 LUISA_ERROR_WITH_LOCATION(
58 "Invalid mipmap level {} for volume with {} levels.",
59 level, _mip_levels);
60 }
61 auto mip_size = luisa::max(_size >> level, 1u);
62 return VolumeView<T>{handle(), _storage, level, {}, mip_size};
63 }
64
65 [[nodiscard]] auto region(uint3 offset, uint3 size) const noexcept {
66 return this->view().region(offset, size);
67 }
68
69 template<typename UVW>

72 }
73

83 template<typename U>
