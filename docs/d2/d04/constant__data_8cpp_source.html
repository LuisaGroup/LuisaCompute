31 using namespace std::string_view_literals;
32 auto hash = hash64(type->hash(), hash64("__hash_constant_data"sv));
33 hash = luisa::detail::xxh3_hash64(view.data(), view.size_bytes(), hash);
34 std::scoped_lock lock{detail::constant_registry_mutex()};
35 if (auto iter = std::find_if(
36 detail::constant_registry().cbegin(),
37 detail::constant_registry().cend(),
38 [hash](auto &&item) noexcept {
39 return item.first._hash == hash;

41 iter != detail::constant_registry().cend()) { return iter->first; }
42 luisa::vector<std::byte> storage(view.size_bytes());
43 std::memcpy(storage.data(), view.data(), view.size_bytes());
44 luisa::span<const T> new_view{reinterpret_cast<const T *>(storage.data()), view.size()};
45 return detail::constant_registry()
46 .emplace_back(std::make_pair(
47 ConstantData{new_view, hash}, std::move(storage)))
