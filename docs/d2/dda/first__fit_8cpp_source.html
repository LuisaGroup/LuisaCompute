48FirstFit::Node *FirstFit::allocate(size_t size) noexcept {
49 // walk the free list
50 for (auto p = &_free_list; p->_next != nullptr; p = p->_next) {
51 // found available node
52 if (auto node = p->_next; node->_size > size) {
53 // compute aligned size
54 auto mask = _alignment - 1u;
55 auto aligned_size = (size & mask) == 0u ? size : (size & ~mask) + _alignment;
56 // has remaining size, split the node
57 if (node->_size > aligned_size) {
58 auto alloc_node = detail::first_fit_node_pool().create();
59 alloc_node->_offset = node->_offset;
60 alloc_node->_size = aligned_size;
61 node->_offset += aligned_size;
62 node->_size -= aligned_size;
63 return alloc_node;
64 }
65 // no more remaining size, use the whole node
66 p->_next = node->_next;
67 return node;
68 }
69 }
70 // none available
71 return nullptr;
72}

74void FirstFit::free(FirstFit::Node *node) noexcept {
75 if (node != nullptr) [[likely]] {
76 auto first = _free_list._next;
77 size_t node_end = node->_offset + node->_size;
78 if (first == nullptr || node_end < first->_offset) {// insert as first, no merge
79 node->_next = first;
80 _free_list._next = node;
81 return;
82 }
83 if (node_end == first->_offset) {// insert as first, merge
84 first->_offset = node->_offset;
85 first->_size += node->_size;
86 detail::first_fit_node_pool().recycle(node);
87 return;
88 }
89 // should not be the first node
90 for (auto p = first; p != nullptr; p = p->_next) {
91 // found the node after which we should insert...
92 auto next = p->_next;
93 auto prev_end = p->_offset + p->_size;
94 if (prev_end < node->_offset) {// no merge with prev
95 // no merge with next
96 if (next == nullptr || node_end < next->_offset) {
97 node->_next = p->_next;
98 p->_next = node;
99 return;
100 }
101 // merge with next
102 if (node_end == next->_offset) {
103 next->_offset = node->_offset;
104 next->_size += node->_size;
105 detail::first_fit_node_pool().recycle(node);
106 return;
107 }
108 } else if (prev_end == node->_offset) {// merge with prev
109 // no merge with next
110 if (next == nullptr || node_end < next->_offset) {
111 p->_size += node->_size;
112 detail::first_fit_node_pool().recycle(node);
113 return;
114 }
115 // merge with prev & next
116 if (node_end == next->_offset) {
117 p->_size += node->_size + next->_size;
118 p->_next = next->_next;
119 detail::first_fit_node_pool().recycle(node);
120 detail::first_fit_node_pool().recycle(next);
121 return;
122 }
123 }
124 }
125 LUISA_ERROR_WITH_LOCATION(
126 "Invalid node for first-fit free list "
127 "(offset = {}, size = {}). Free list dump: {}.",
128 node->_offset, node->_size, dump_free_list());
129 }
130}

132inline void FirstFit::_destroy() noexcept {
133 if (_free_list._size != 0u) {
134 if (_free_list._next == nullptr ||
135 _free_list._next->_next != nullptr ||
136 _free_list._next->_offset != 0u ||
137 _free_list._next->_size != _free_list._size) [[unlikely]] {
138 LUISA_WARNING_WITH_LOCATION("Leaks in first-fit free list.");
139 }
140 auto p = _free_list._next;
141 while (p != nullptr) {
142 auto node = p;
143 p = p->_next;
144 detail::first_fit_node_pool().recycle(node);
145 }
146 }
147}
