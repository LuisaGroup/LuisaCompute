10ISPCAccel::ISPCAccel(RTCDevice device, AccelBuildHint hint) noexcept
11 : _handle{rtcNewScene(device)} {
12 switch (hint) {
13 case AccelBuildHint::FAST_TRACE:
14 rtcSetSceneBuildQuality(_handle, RTC_BUILD_QUALITY_HIGH);
15 rtcSetSceneFlags(_handle, RTC_SCENE_FLAG_COMPACT);
16 break;
17 case AccelBuildHint::FAST_UPDATE:
18 rtcSetSceneBuildQuality(_handle, RTC_BUILD_QUALITY_MEDIUM);
19 rtcSetSceneFlags(_handle, RTC_SCENE_FLAG_COMPACT | RTC_SCENE_FLAG_DYNAMIC);
20 break;
21 case AccelBuildHint::FAST_REBUILD:
22 rtcSetSceneBuildQuality(_handle, RTC_BUILD_QUALITY_LOW);
23 rtcSetSceneFlags(_handle, RTC_SCENE_FLAG_DYNAMIC);
24 break;
25 }
26}

35void ISPCAccel::build(ThreadPool &pool) noexcept {
36 _dirty.clear();
37 _resources.clear();
38 for (auto &_instance : _instances) {
39 auto mesh = _instance.mesh;
40 _resources.emplace(mesh->vertex_buffer());
41 _resources.emplace(mesh->triangle_buffer());
42 _resources.emplace(reinterpret_cast<uint64_t>(mesh));
43 }
44 pool.async([this, instances = _instances] {
45 for (auto i = 0u; i < _committed_geometries.size(); i++) {
46 rtcDetachGeometry(_handle, i);
47 rtcReleaseGeometry(_committed_geometries[i]);
48 }
49 auto device = rtcGetSceneDevice(_handle);
50 _committed_transforms.resize(instances.size());
51 _committed_geometries.resize(instances.size());
52 for (auto i = 0u; i < instances.size(); i++) {
53 auto transform = instances[i].transform;
54 _committed_transforms[i] = _decompress(transform);
55 auto geometry = rtcNewGeometry(device, RTC_GEOMETRY_TYPE_INSTANCE);
56 _committed_geometries[i] = geometry;
57 rtcSetGeometryInstancedScene(geometry, instances[i].mesh->handle());
58 rtcSetGeometryBuildQuality(geometry, RTC_BUILD_QUALITY_HIGH);
59 rtcSetGeometryTransform(
60 geometry, 0.f,
61 RTC_FORMAT_FLOAT4X4_COLUMN_MAJOR,
62 &_committed_transforms[i]);
63 if (!instances[i].visible) { rtcDisableGeometry(geometry); }
64 rtcCommitGeometry(geometry);
65 rtcAttachGeometryByID(_handle, geometry, i);
66 }
67 rtcCommitScene(_handle);
68 });
69}

71void ISPCAccel::update(ThreadPool &pool) noexcept {
72 auto s = luisa::span{_instances}.subspan(_dirty.offset(), _dirty.size());
73 pool.async([this, offset = _dirty.offset(),
74 instances = luisa::vector<Instance>{s.cbegin(), s.cend()}] {
75 for (auto i = 0u; i < instances.size(); i++) {
76 auto geometry = _committed_geometries[i + offset];
77 if (instances[i].visible) {
78 rtcEnableGeometry(geometry);
79 } else {
80 rtcDisableGeometry(geometry);
81 }
82 auto transform = instances[i].transform;
83 _committed_transforms[i + offset] = _decompress(transform);
84 rtcSetGeometryTransform(
85 geometry, 0.f, RTC_FORMAT_FLOAT4X4_COLUMN_MAJOR,
86 &_committed_transforms[i + offset]);
87 rtcCommitGeometry(geometry);
88 }
89 rtcCommitScene(_handle);
90 });
91 _dirty.clear();
92}
