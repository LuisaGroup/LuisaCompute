32void ISPCAccel::build(ThreadPool &pool, size_t instance_count,
33 luisa::span<const AccelBuildCommand::Modification> mods) noexcept {
34
35 using Mod = AccelBuildCommand::Modification;
36 pool.async([this, instance_count, mods = luisa::vector<Mod>{mods.cbegin(), mods.cend()}]() {
37 if (instance_count < _instances.size()) {// remove redundant geometries
38 for (auto i = instance_count; i < _instances.size(); i++) { rtcDetachGeometry(_handle, i); }
39 _instances.resize(instance_count);
40 } else {// create new geometries
41 auto device = rtcGetSceneDevice(_handle);
42 _instances.reserve(next_pow2(instance_count));
43 for (auto i = _instances.size(); i < instance_count; i++) {
44 auto geometry = rtcNewGeometry(device, RTC_GEOMETRY_TYPE_INSTANCE);
45 rtcSetGeometryBuildQuality(geometry, RTC_BUILD_QUALITY_HIGH);
46 rtcAttachGeometryByID(_handle, geometry, i);
47 rtcReleaseGeometry(geometry);// already moved into the scene
48 _instances.emplace_back().geometry = geometry;
49 }
50 }
51 for (auto m : mods) {
52 auto geometry = _instances[m.index].geometry;
53 if (m.flags & Mod::flag_mesh) { rtcSetGeometryInstancedScene(
54 geometry, reinterpret_cast<const ISPCMesh *>(m.mesh)->handle()); }
55 if (m.flags & Mod::flag_transform) { std::memcpy(
56 _instances[m.index].affine, m.affine, sizeof(m.affine)); }
57 if (m.flags & Mod::flag_visibility_on) { _instances[m.index].visible = true; }
58 if (m.flags & Mod::flag_visibility_off) { _instances[m.index].visible = false; }
59 _instances[m.index].dirty = true;
60 }
61 for (auto &&instance : _instances) {
62 if (instance.dirty) {
63 auto geometry = instance.geometry;
64 rtcSetGeometryTransform(geometry, 0u, RTC_FORMAT_FLOAT3X4_ROW_MAJOR, instance.affine);
65 instance.visible ? rtcEnableGeometry(geometry) : rtcDisableGeometry(geometry);
66 rtcCommitGeometry(geometry);
67 instance.dirty = false;
68 }
69 }
70 rtcCommitScene(_handle);
71 });
72}
