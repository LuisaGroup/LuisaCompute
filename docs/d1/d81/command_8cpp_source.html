14inline void ShaderDispatchCommand::_encode_buffer(uint64_t handle, size_t offset) noexcept {
15 if (_argument_count >= _kernel.arguments().size()) [[unlikely]] {
16 LUISA_ERROR_WITH_LOCATION(
17 "Invalid buffer argument at index {}.",
18 _argument_count);
19 }
20 if (_argument_buffer_size + sizeof(BufferArgument) > _argument_buffer.size()) [[unlikely]] {
21 LUISA_ERROR_WITH_LOCATION(
22 "Failed to encode buffer. "
23 "Shader argument buffer exceeded size limit {}.",
24 _argument_buffer.size());
25 }
26 if (auto t = _kernel.arguments()[_argument_count].type();
27 !t->is_buffer()) {
28 LUISA_ERROR_WITH_LOCATION(
29 "Expected {} but got buffer for argument {}.",
30 t->description(), _argument_count);
31 }
32 auto variable_uid = _kernel.arguments()[_argument_count].uid();
33 auto usage = _kernel.variable_usage(variable_uid);
34 BufferArgument argument{variable_uid, handle, offset};
35 std::memcpy(
36 _argument_buffer.data() + _argument_buffer_size,
37 &argument, sizeof(BufferArgument));
38 _argument_buffer_size += sizeof(BufferArgument);
39 _argument_count++;
40}

169inline void ShaderDispatchCommand::_encode_pending_bindings() noexcept {
170 auto bindings = _kernel.builder()->argument_bindings();
171 while (_argument_count < _kernel.arguments().size() &&
172 !luisa::holds_alternative<luisa::monostate>(bindings[_argument_count])) {
173 luisa::visit(
174 [&, arg = _kernel.arguments()[_argument_count]]<typename T>(T binding) noexcept {
175 if constexpr (std::is_same_v<T, detail::FunctionBuilder::BufferBinding>) {
176 _encode_buffer(binding.handle, binding.offset_bytes);
177 } else if constexpr (std::is_same_v<T, detail::FunctionBuilder::TextureBinding>) {
178 _encode_texture(binding.handle, binding.level);
179 } else if constexpr (std::is_same_v<T, detail::FunctionBuilder::BindlessArrayBinding>) {
180 _encode_bindless_array(binding.handle);
181 } else if constexpr (std::is_same_v<T, detail::FunctionBuilder::AccelBinding>) {
182 _encode_accel(binding.handle);
183 } else {
184 LUISA_ERROR_WITH_LOCATION("Invalid argument binding type.");
185 }
186 },
187 bindings[_argument_count]);
188 }
189}
