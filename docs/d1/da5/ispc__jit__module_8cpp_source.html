37[[nodiscard]] auto get_target_machine() {
38 static std::once_flag flag;
39 std::call_once(flag, [] {
40 llvm::InitializeNativeTarget();
41 llvm::InitializeNativeTargetAsmPrinter();
42 LLVMLinkInMCJIT();
43 });
44 std::string err;
45 auto target_triple = llvm::sys::getDefaultTargetTriple();
46 LUISA_INFO("Target: {}.", target_triple);
47 auto target = llvm::TargetRegistry::lookupTarget(target_triple, err);
48 if (target == nullptr) {
49 LUISA_ERROR_WITH_LOCATION("Failed to get target machine: {}.", err);
50 }
51 llvm::TargetOptions options;
52 options.AllowFPOpFusion = llvm::FPOpFusion::Fast;
53 options.UnsafeFPMath = true;
54 options.NoInfsFPMath = true;
55 options.NoNaNsFPMath = true;
56 options.NoTrappingFPMath = true;
57 options.GuaranteedTailCallOpt = true;
58 auto mcpu = llvm::sys::getHostCPUName();
59 auto machine = target->createTargetMachine(
60 target_triple, mcpu,
61#if defined(LUISA_PLATFORM_APPLE) && defined(__aarch64__)
62 "+neon",
63#else
64 "+avx2",
65#endif
66 options, {}, {},
67 llvm::CodeGenOpt::Aggressive, true);
68 if (machine == nullptr) {
69 LUISA_ERROR_WITH_LOCATION("Failed to create target machine.");
70 }
71 return machine;
72}

74luisa::shared_ptr<ISPCModule> ISPCJITModule::load(
75 const Context &ctx, const std::filesystem::path &ir_path) noexcept {
76 // load
77 auto context = luisa::make_unique<llvm::LLVMContext>();
78 llvm::SMDiagnostic error;
79 LUISA_INFO("Loading LLVM IR: '{}'.", ir_path.string());
80 auto module = llvm::parseIRFile(ir_path.string(), error, *context);
81 if (module == nullptr) {
82 LUISA_ERROR_WITH_LOCATION(
83 "Failed to load module: {}.",
84 error.getMessage().data());
85 }
86
87 // optimize: machine
88 auto machine = get_target_machine();
89 llvm::PassManagerBuilder pass_manager_builder;
90 pass_manager_builder.OptLevel = llvm::CodeGenOpt::Aggressive;
91 pass_manager_builder.Inliner = llvm::createFunctionInliningPass(
92 pass_manager_builder.OptLevel, 0, false);
93 pass_manager_builder.LoopsInterleaved = true;
94 pass_manager_builder.LoopVectorize = true;
95 pass_manager_builder.SLPVectorize = true;
96 pass_manager_builder.MergeFunctions = true;
97 pass_manager_builder.EnablePGOCSInstrGen = false;
98 pass_manager_builder.EnablePGOCSInstrUse = false;
99 pass_manager_builder.EnablePGOInstrGen = false;
100 pass_manager_builder.CallGraphProfile = false;
101 pass_manager_builder.PerformThinLTO = true;
102 machine->adjustPassManager(pass_manager_builder);
103 module->setDataLayout(machine->createDataLayout());
104
105 // optimize: function passes
106 {
107 llvm::legacy::FunctionPassManager pass_manager{module.get()};
108 pass_manager_builder.populateFunctionPassManager(pass_manager);
109 pass_manager.add(llvm::createTargetTransformInfoWrapperPass(
110 machine->getTargetIRAnalysis()));
111 pass_manager_builder.populateFunctionPassManager(pass_manager);
112 pass_manager.doInitialization();
113 for (auto &&f : module->functions()) {
114 pass_manager.run(f);
115 }
116 pass_manager.doFinalization();
117 }
118
119 // optimize: module passes
120 {
121 llvm::legacy::PassManager pass_manager;
122 pass_manager.add(
123 llvm::createTargetTransformInfoWrapperPass(machine->getTargetIRAnalysis()));
124 pass_manager_builder.populateModulePassManager(pass_manager);
125 pass_manager.run(*module);
126 }
127
128 // jit
129 std::string err;
130 std::unique_ptr<llvm::ExecutionEngine> engine{
131 llvm::EngineBuilder{std::move(module)}
132 .setErrorStr(&err)
133 .setOptLevel(llvm::CodeGenOpt::Aggressive)
134 .setEngineKind(llvm::EngineKind::JIT)
135 .create(machine)};
136 engine->DisableLazyCompilation(true);
137 engine->DisableSymbolSearching(false);// to support print in ispc
138 engine->addGlobalMapping("fflush", reinterpret_cast<uint64_t>(&fflush));
139 engine->addGlobalMapping("fputs", reinterpret_cast<uint64_t>(&fputs));
140 engine->addGlobalMapping("snprintf", reinterpret_cast<uint64_t>(&snprintf));
141 engine->addGlobalMapping("vsnprintf", reinterpret_cast<uint64_t>(&vsnprintf));
142 if (engine == nullptr) {
143 LUISA_ERROR_WITH_LOCATION(
144 "Failed to create execution engine: {}.",
145 err);
146 }
147 return luisa::make_shared<ISPCJITModule>(
148 ISPCJITModule{std::move(context), std::move(engine)});
149}
