32 reinterpret_cast<const void *>(_t_buffer),
33 _t_offset, sizeof(Triangle), _t_count);
34 switch (_hint) {
35 case AccelBuildHint::FAST_TRACE:
36 rtcSetGeometryBuildQuality(_geometry, RTC_BUILD_QUALITY_HIGH);

40 rtcSetGeometryBuildQuality(_geometry, RTC_BUILD_QUALITY_REFIT);
41 break;
42 }

44 rtcSetSceneBuildQuality(_handle, RTC_BUILD_QUALITY_HIGH);

46 }
