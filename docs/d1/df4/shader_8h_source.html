151class Shader final : public Resource {
152
153 static_assert(dimension == 1u || dimension == 2u || dimension == 3u);
154
155private:

158private:

160 Shader(Device::Interface *device,
161 luisa::shared_ptr<const detail::FunctionBuilder> kernel,
162 std::string_view meta_options) noexcept
163 : Resource{device, Tag::SHADER, device->create_shader(kernel->function(), meta_options)},
164 _kernel{std::move(kernel)} {}
165
166public:
167 Shader() noexcept = default;
168 using Resource::operator bool;
169 [[nodiscard]] auto operator()(detail::prototype_to_shader_invocation_t<Args>... args) const noexcept {
170 using invoke_type = detail::ShaderInvoke<dimension>;
171 invoke_type invoke{handle(), _kernel->function()};
172 return static_cast<invoke_type &&>((invoke << ... << args));
173 }
174};
