42class ShaderInvokeBase {
43
44private:
45 Command *_command;
46 Function _kernel;
47 size_t _argument_index{0u};
48
49private:
50 [[nodiscard]] auto _dispatch_command() noexcept {
51 return static_cast<ShaderDispatchCommand *>(_command);
52 }
53 void _encode_pending_bindings() noexcept {
54 auto bindings = _kernel.builder()->argument_bindings();
55 for (; _argument_index < _kernel.arguments().size() &&
56 !luisa::holds_alternative<luisa::monostate>(// has binding
57 bindings[_argument_index]);
58 _argument_index++) {
59 luisa::visit(
60 [&, arg = _kernel.arguments()[_argument_index]]<typename T>(T binding) noexcept {
61 if constexpr (std::is_same_v<T, FunctionBuilder::BufferBinding>) {
62 _dispatch_command()->encode_buffer(
63 arg.uid(), binding.handle, binding.offset_bytes,
64 _kernel.variable_usage(arg.uid()));
65 } else if constexpr (std::is_same_v<T, FunctionBuilder::TextureBinding>) {
66 _dispatch_command()->encode_texture(
67 arg.uid(), binding.handle, binding.level,
68 _kernel.variable_usage(arg.uid()));
69 } else if constexpr (std::is_same_v<T, FunctionBuilder::BindlessArrayBinding>) {
70 _dispatch_command()->encode_bindless_array(arg.uid(), binding.handle);
71 } else if constexpr (std::is_same_v<T, FunctionBuilder::AccelBinding>) {
72 _dispatch_command()->encode_accel(arg.uid(), binding.handle);
73 } else {
74 LUISA_ERROR_WITH_LOCATION("Invalid argument binding type.");
75 }
76 },
77 bindings[_argument_index]);
78 }
79 }
80
81public:
82 explicit ShaderInvokeBase(uint64_t handle, Function kernel) noexcept
83 : _command{ShaderDispatchCommand::create(handle, kernel)},
84 _kernel{kernel} {}
85 ShaderInvokeBase(ShaderInvokeBase &&) noexcept = default;
86 ShaderInvokeBase(const ShaderInvokeBase &) noexcept = delete;
87 ShaderInvokeBase &operator=(ShaderInvokeBase &&) noexcept = default;
88 ShaderInvokeBase &operator=(const ShaderInvokeBase &) noexcept = delete;

90 template<typename T>
91 ShaderInvokeBase &operator<<(BufferView<T> buffer) noexcept {
92 _encode_pending_bindings();
93 if (auto t = _kernel.arguments()[_argument_index].type();
94 !t->is_buffer()) {
95 LUISA_ERROR_WITH_LOCATION(
96 "Expected {} but got buffer for argument {}.",
97 t->description(), _argument_index);
98 }
99 auto variable_uid = _kernel.arguments()[_argument_index++].uid();
100 auto usage = _kernel.variable_usage(variable_uid);
101 _dispatch_command()->encode_buffer(
102 variable_uid, buffer.handle(), buffer.offset_bytes(), usage);
103 return *this;
104 }

151 template<typename T>
152 ShaderInvokeBase &operator<<(T data) noexcept {
153 _encode_pending_bindings();
154 if (auto t = _kernel.arguments()[_argument_index].type();
155 (!t->is_basic() && !t->is_structure() && !t->is_array()) ||
156 t->size() != sizeof(T)) {
157 LUISA_ERROR_WITH_LOCATION(
158 "Invalid uniform (size = {}) at index {}, "
159 "expected {} (size = {}).",
160 sizeof(T), _argument_index,
161 t->description(), t->size());
162 }
163 auto variable_uid = _kernel.arguments()[_argument_index++].uid();
164 _dispatch_command()->encode_uniform(variable_uid, &data, sizeof(T), alignof(T));
165 return *this;
166 }

174protected:
175 [[nodiscard]] auto _parallelize(uint3 dispatch_size) &&noexcept {
176 _encode_pending_bindings();
177 _dispatch_command()->set_dispatch_size(dispatch_size);
178 Command *command{nullptr};
179 std::swap(command, _command);
180 return command;
181 }

221template<size_t dimension, typename... Args>
222class Shader final : public Resource {
223
224 static_assert(dimension == 1u || dimension == 2u || dimension == 3u);
225
226private:
227 luisa::shared_ptr<const detail::FunctionBuilder> _kernel;
228
229private:
230 friend class Device;
231 Shader(Device::Interface *device, luisa::shared_ptr<const detail::FunctionBuilder> kernel, std::string_view meta_options) noexcept
232 : Resource{
233 device,
234 Tag::SHADER,
235 device->create_shader(kernel->function(), meta_options)},
236 _kernel{std::move(kernel)} {}
237
238public:
239 Shader() noexcept = default;
240 using Resource::operator bool;
241 [[nodiscard]] auto operator()(detail::prototype_to_shader_invocation_t<Args>... args) const noexcept {
242 using invoke_type = detail::ShaderInvoke<dimension>;
243 invoke_type invoke{handle(), _kernel->function()};
244 return static_cast<invoke_type &&>((invoke << ... << args));
245 }
246};
