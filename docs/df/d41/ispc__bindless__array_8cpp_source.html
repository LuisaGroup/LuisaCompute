57void ISPCBindlessArray::update(ThreadPool &pool) noexcept {
58 auto s = luisa::span{_slots};//.subspan(_dirty.offset(), _dirty.size());
59 pool.async([this, offset = 0u,//offset = _dirty.offset(),
60 slots = luisa::vector<Slot>{s.cbegin(), s.cend()}] {
61 for (auto i = 0u; i < slots.size(); i++) {
62 auto slot = slots[i];
63 if (auto b = static_cast<const std::byte *>(slot.buffer)) {
64 _items[i + offset].buffer = b + slot.buffer_offset;
65 }
66 if (auto t = slot.tex2d) {
67 _items[i + offset].tex2d = t->handle();
68 _items[i + offset].sampler2d = slot.sampler2d.code();
69 }
70 if (auto t = slot.tex3d) {
71 _items[i + offset].tex3d = t->handle();
72 _items[i + offset].sampler3d = slot.sampler3d.code();
73 }
74 }
75 });
76 _dirty.clear();
77 _tracker.commit();
78}
