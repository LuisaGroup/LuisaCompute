31// reference: https://github.com/yohhoy/yamc/blob/master/include/yamc_barrier.hpp
32class Barrier {
33
34private:
35 uint _n;
36 uint _counter;
37 uint _phase;
38 std::condition_variable _cv;
39 std::mutex _mutex;
40
41public:
42 explicit Barrier(uint n) noexcept
43 : _n{n}, _counter{n}, _phase{0u} {}
44 void arrive_and_wait() noexcept {
45 std::unique_lock lock{_mutex};
46 auto arrive_phase = _phase;
47 if (--_counter == 0u) {
48 _counter = _n;
49 _phase++;
50 _cv.notify_all();
51 }
52 while (_phase <= arrive_phase) {
53 _cv.wait(lock);
54 }
55 }
56};

58ThreadPool::ThreadPool(size_t num_threads) noexcept : _should_stop{false} {
59 if (num_threads == 0u) {
60 num_threads = std::max(
61 std::thread::hardware_concurrency(), 1u);
62 }
63 _dispatch_barrier = luisa::make_unique<Barrier>(num_threads);
64 _synchronize_barrier = luisa::make_unique<Barrier>(num_threads + 1u /* main thread */);
65 _threads.reserve(num_threads);
66 for (auto i = 0u; i < num_threads; i++) {
67 _threads.emplace_back(std::thread{[this] {
68 detail::is_worker_thread() = true;
69 for (;;) {
70 std::unique_lock lock{_mutex};
71 _cv.wait(lock, [this] { return !_tasks.empty() || _should_stop; });
72 if (_should_stop && _tasks.empty()) [[unlikely]] { break; }
73 auto task = std::move(_tasks.front());
74 _tasks.pop();
75 lock.unlock();
76 task();
77 }
78 }});
79 }
80 LUISA_INFO(
81 "Created thread pool with {} thread{}.",
82 num_threads, num_threads == 1u ? "" : "s");
83}

104void ThreadPool::_dispatch_all(luisa::function<void()> task, size_t max_threads) noexcept {
105 {
106 std::scoped_lock lock{_mutex};
107 for (auto i = 0u; i < std::min(_threads.size(), max_threads) - 1u; i++) {
108 _tasks.emplace(task);
109 }
110 _tasks.emplace(std::move(task));
111 }
112 _cv.notify_all();
113}
