252template<typename S, typename Members, typename offsets>
253struct is_valid_reflection : std::false_type {};
254
255template<typename S, typename... M, typename O, O... os>
256struct is_valid_reflection<S, std::tuple<M...>, std::integer_sequence<O, os...>> {
257
258 static_assert(((!is_struct_v<M> || alignof(M) >= 4u) && ...));
259 static_assert((!is_bool_vector_v<M> && ...),
260 "Boolean vectors are not allowed in DSL "
261 "structures since their may have different "
262 "layouts on different platforms.");
263
264private:
265 [[nodiscard]] constexpr static auto _check() noexcept {
266 constexpr auto count = sizeof...(M);
267 static_assert(sizeof...(os) == count);
268 constexpr std::array<size_t, count> sizes{sizeof(M)...};
269 constexpr std::array<size_t, count> alignments{alignof(M)...};
270 constexpr std::array<size_t, count> offsets{os...};
271 auto current_offset = 0u;
272 for (auto i = 0u; i < count; i++) {
273 auto offset = offsets[i];
274 auto size = sizes[i];
275 auto alignment = alignments[i];
276 current_offset = (current_offset + alignment - 1u) /
277 alignment *
278 alignment;
279 if (current_offset != offset) { return false; }
280 current_offset += size;
281 }
282 constexpr auto struct_size = sizeof(S);
283 constexpr auto struct_alignment = alignof(S);
284 current_offset = (current_offset + struct_alignment - 1u) /
285 struct_alignment *
286 struct_alignment;
287 return current_offset == struct_size;
288 };
289
290public:
291 static constexpr auto value = _check();
292};
293
294template<typename S, typename M, typename O>
295constexpr auto is_valid_reflection_v = is_valid_reflection<S, M, O>::value;
