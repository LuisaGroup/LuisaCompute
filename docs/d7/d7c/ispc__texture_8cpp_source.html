10ISPCTexture::ISPCTexture(PixelFormat format, uint dim, uint3 size, uint mip_levels) noexcept : storage(pixel_format_to_storage(format)),
11 dim(dim),
12 lodLevel(mip_levels) {
13 if (dim == 2) size.z = 1;
14 this->size[0] = size.x;
15 this->size[1] = size.y;
16 this->size[2] = size.z;
17 if (dim != 2 && dim != 3) LUISA_ERROR_WITH_LOCATION("Only 2D / 3D texture is supported");
18 if (lodLevel > MAXLOD) LUISA_ERROR_WITH_LOCATION("maximal LoD exceeded");
19 if (lodLevel == 0) LUISA_ERROR_WITH_LOCATION("LoD must be at least 1");
20 // mipmap allocate
21 uint pxsize = pixel_storage_size(storage);
22 int offset[MAXLOD + 1];
23 offset[0] = 0;
24 for (int i = 0; i < lodLevel; ++i) {
25 uint w = std::max(size.x >> i, 1u);
26 uint h = std::max(size.y >> i, 1u);
27 uint d = std::max(size.z >> i, 1u);
28 offset[i + 1] = offset[i] + w * h * d * pxsize;
29 }
30 lods[0] = allocate<uint8_t>(offset[lodLevel]);
31 LUISA_VERBOSE_WITH_LOCATION(
32 "New float array for ISPC with {} bytes.",
33 offset[lodLevel]);
34 for (int i = 1; i < lodLevel; ++i) {
35 lods[i] = (void *)((uint8_t *)lods[0] + offset[i]);
36 }
37}
