34[[nodiscard]] luisa::string win32_last_error_message() {
35 // Retrieve the system error message for the last-error code
36 void *buffer = nullptr;
37 auto err_code = GetLastError();
38 FormatMessage(
39 FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
40 nullptr,
41 err_code,
42 MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
43 (LPTSTR)&buffer,
44 0, nullptr);
45 luisa::string err_msg{fmt::format("{} (code = 0x{:x}).", static_cast<char *>(buffer), err_code)};
46 LocalFree(buffer);
47 return err_msg;
48}

100luisa::vector<TraceItem> backtrace() noexcept {
101
102 void *stack[100];
103 auto process = GetCurrentProcess();
104 SymInitialize(process, nullptr, true);
105 auto frame_count = CaptureStackBackTrace(0, 100, stack, nullptr);
106
107 struct Symbol : SYMBOL_INFO {
108 char name_storage[1023];
109 } symbol{};
110 symbol.MaxNameLen = 1024;
111 symbol.SizeOfStruct = sizeof(SYMBOL_INFO);
112 IMAGEHLP_MODULE64 module{};
113 module.SizeOfStruct = sizeof(IMAGEHLP_MODULE64);
114 luisa::vector<TraceItem> trace;
115 trace.reserve(frame_count - 1u);
116 for (auto i = 1u; i < frame_count; i++) {
117 auto address = reinterpret_cast<uint64_t>(stack[i]);
118 auto displacement = 0ull;
119 if (SymFromAddr(process, address, &displacement, &symbol)) {
120 TraceItem item{};
121 if (SymGetModuleInfo64(process, symbol.ModBase, &module)) {
122 item.module = module.ModuleName;
123 } else {
124 item.module = "???";
125 }
126 item.symbol = symbol.Name;
127 item.address = address;
128 item.offset = displacement;
129 trace.emplace_back(std::move(item));
130 } else {
131 LUISA_VERBOSE_WITH_LOCATION(
132 "Failed to get stacktrace at 0x{:012}: {}",
133 address, detail::win32_last_error_message());
134 }
135 }
136 return trace;
137}

158void *dynamic_module_load(const std::filesystem::path &path) noexcept {
159 auto p = path;
160 for (auto ext : {".so", ".dylib"}) {
161 p.replace_extension(ext);
162 if (auto module = dlopen(p.c_str(), RTLD_LAZY); module != nullptr) {
163 return module;
164 }
165 LUISA_WARNING_WITH_LOCATION(
166 "Failed to load dynamic module '{}', reason: {}.",
167 p.string(), dlerror());

205luisa::vector<TraceItem> backtrace() noexcept {
206 void *trace[100u];
207 auto count = ::backtrace(trace, 100);
208 auto info = ::backtrace_symbols(trace, count);
209 luisa::vector<TraceItem> trace_info;
210 trace_info.reserve(count - 1u);
211 for (auto i = 1 /* skip current frame */; i < count; i++) {
212 std::istringstream iss{info[i]};
213 auto index = 0;
214 char plus = '+';
215 TraceItem item{};
216 iss >> index >> item.module >> std::hex >> item.address >> item.symbol >> plus >> std::dec >> item.offset;
217 item.symbol = demangle(item.symbol.c_str());
218 trace_info.emplace_back(std::move(item));
219 }
220 free(info);
221 return trace_info;
222}
